	id = pdengine.07
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes

		has_country_flag = pdtidallylockedhomeworldmoon 
		any_owned_planet = {
			is_planet_class = pc_tidallylocked
		}
	}

	immediate = {
		random_planet_within_border = {
			limit = { is_capital = no is_moon = no is_star = no is_asteroid = no NOT = { is_planet_class = pc_gas_giant } habitable_structure =  no }
			set_planet_size = 20
			change_pc = pc_tidallylocked
			generate_pd_start_deposits_and_blockers = yes
			#add_modifier = {
			#	modifier = "tidal_locked2"
			#	days = -1
			#}
			set_owner = ROOT
			set_planet_flag = tidallylockedhomeworld
			#All other pop types
			create_pop = { 
				species = owner_species
				ethos = random
			}	
			set_capital = yes
			generate_start_buildings_and_districts = yes
			generate_start_pops = yes
			IF = {
				limit = { owner = { has_origin = origin_tree_of_life } }
