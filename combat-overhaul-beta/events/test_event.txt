event = {
	id = test_event.1
	hide_window = yes
	is_triggered_only = yes	

	immediate = {
		every_country = {
			limit = { 
				AND = {
					max_naval_capacity > 100
					max_naval_capacity < 200
				}	
			}
			add_modifier = { 
				modifier = "test_modifier" 
				days = -1 
			}	
		}
	}	
}

				num_pops < 200

	}
