check_neighboring_adj_bonus_buildings_tycoon = {
	if = {
		limit = {
			NAND = {
				exists = owner
				owner = { has_authority = auth_machine_intelligence }
			}
		}
		every_neighboring_tile = {
			if = {
				limit = { has_building = building_bank_1 }
				prevprev = {
					change_variable = { which = energy_weight value = 1 }
				}
				else = {
					if = {
						limit = { has_building = building_bank_2 }
						prevprev = {
							change_variable = { which = energy_weight value = 1 }
						}
						else = {
							if = {
								limit = { has_building = building_bank_3 }
								prevprev = {
									change_variable = { which = energy_weight value = 1 }
								}
								else = {
									if = {
										limit = { has_building = building_bank_4 }
										prevprev = {
											change_variable = { which = energy_weight value = 2 }
										}
										else = {
											if = {
												limit = { has_building = building_bank_5 }
												prevprev = {
													change_variable = { which = energy_weight value = 2 }
												}
												else = {
													if = {
														limit = { has_building = building_central_bank_6 }
														prevprev = {
															change_variable = { which = energy_weight value = 5 }
														}
													}
												}
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}
	}
	if = {
		limit = {
			NAND = {
				exists = owner
				owner = { NOT = { has_authority = auth_machine_intelligence } }
			}
		}
		every_neighboring_tile = {
			if = {
				limit = { has_building = building_battery_array_1 }
				prevprev = {
					change_variable = { which = energy_weight value = 1 }
				}
				else = {
					if = {
						limit = { has_building = building_battery_array_2 }
						prevprev = {
							change_variable = { which = energy_weight value = 1 }
						}
						else = {
							if = {
								limit = { has_building = building_battery_array_3 }
								prevprev = {
									change_variable = { which = energy_weight value = 1 }
								}
								else = {
									if = {
										limit = { has_building = building_battery_array_4 }
										prevprev = {
											change_variable = { which = energy_weight value = 2 }
										}
										else = {
											if = {
												limit = { has_building = building_battery_array_5 }
												prevprev = {
													change_variable = { which = energy_weight value = 2 }
												}
												else = {
													if = {
														limit = { has_building = building_main_battery_array_6 }
														prevprev = {
															change_variable = { which = energy_weight value = 5 }
														}
													}
												}
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}
	}
	every_neighboring_tile = {
		if = {
			limit = { has_building = building_industrial_complex_1 }
			prevprev = {
				change_variable = { which = minerals_weight value = 4 }
			}
			else = {
				if = {
					limit = { has_building = building_industrial_complex_2 }
					prevprev = {
						change_variable = { which = minerals_weight value = 6 }
					}
					else = {
						if = {
							limit = { has_building = building_industrial_complex_3 }
							prevprev = {
								change_variable = { which = minerals_weight value = 8 }
							}
						}
					}
				}
			}
		}
	}
}
