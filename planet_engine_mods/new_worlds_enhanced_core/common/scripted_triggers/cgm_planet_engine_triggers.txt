deposit_and_blocker_threshold_limit = { 
	OR = { 
		AND = {
			planet_size > 19
			NOT = { 
				num_pops > 12
			}	
			count_tile = {
				limit = { has_blocker = yes }
				count < 6
			}
		}
		AND = {
			planet_size > 15
			planet_size < 19
			NOT = { 
				num_pops > 10
			}						
			count_tile = {
				limit = { has_blocker = yes }
				count < 5
			}
		}
		AND = {
			planet_size > 10
			planet_size < 16
			NOT = { 
				num_pops > 8
			}										
			count_tile = {
				limit = { has_blocker = yes }
				count < 4
			}
		}
		AND = {
			planet_size < 11
			NOT = { 
				num_pops > 5
			}										
			count_tile = {
				limit = { has_blocker = yes }
				count < 3
			}
		}
	}	
}
