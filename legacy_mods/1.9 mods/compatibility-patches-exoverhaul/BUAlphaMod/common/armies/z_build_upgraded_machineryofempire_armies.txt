# Light Mechs
lightmech_army = {
	damage = 2.5
	morale_damage = 1.0
	health = 4.0
	morale = 3.5
	time = 90
	maintenance = 2.5
	icon_frame = 21
	has_species = no
	
	cost = {
		minerals = 200
		energy = 150
	}
	prerequisites = { "tech_alpha_lightmechs" }
	
	allow = {
		custom_tooltip = {
			text = lightmech_troop_limit_tooltip
			OR = {
				has_building = "building_lightmechbay"
				has_building_mediummechbay = yes
				has_building_heavymechbay = yes
				has_building = "building_assaultmechbay"
			}
			planet = {
				NOT = {
					check_variable = {
						which = "lightmech_soldier_count"
						value = 5
					}
				}
			}
		}
	}
	
	on_queued = {
		planet = {
			change_variable = {
				which = "lightmech_soldier_count"
				value = 1
			}
		}
	}
	
	on_unqueued = {
		planet = {
			change_variable = {
				which = "lightmech_soldier_count"
				value = -1
			}
		}
	}
}
