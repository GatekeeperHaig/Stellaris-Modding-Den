cgm_add_special_building_gpm = {
	if = {
		limit = {
			OR = { has_building_construction = yes has_building = yes }
		}
		owner = { set_country_flag = cgm_auto_built }
	}
	else = {
		if = {
			limit = {
				owner = {
					check_variable = { which = cgm_special_bestBuilding value = -2113915.385 }
				}
			}
			add_building_construction = gpm_building_stronghold
		}
	}
}
cgm_search_for_special_building_gpm = {
	if = {
		limit = {
			OR = { has_building_construction = yes has_building = yes }
		}
		owner = { set_country_flag = cgm_auto_built }
	}
	else = {
		if = {
			limit = {
				has_modifier = "Precursor_Bunker_Grid"
				NOR = { has_building = gpm_building_stronghold }
				owner = {
					check_variable = { which = cgm_special_bestWeight value < 10 }
				}
			}
			prev = {
				set_variable = { which = cgm_special_bestWeight value = 10 }
				set_variable = { which = cgm_special_bestBuilding value = -2113915.385 }
			}
			save_event_target_as = cgm_best_planet_for_special
		}
	}
}
