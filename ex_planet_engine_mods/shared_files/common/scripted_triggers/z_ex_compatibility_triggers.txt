###General Triggers

uninhabitable_regular_planet = {
	OR = {
		is_planet_class_pc_molten = yes
		is_planet_class_pc_barren = yes
		is_planet_class_pc_barren_cold = yes
		is_planet_class_pc_toxic = yes
		is_planet_class_pc_frozen = yes	
		AND = { 
			new_worlds_enabled = yes
			new_worlds_uninhabitable_regular_planet = yes
		}
		AND = {
			planetary_diversity_enabled = yes
			planetary_diversity_uninhabitable_regular_planet = yes			
		}			
	}
}

habitable_planet = {
	OR = {
		is_planet_class_pc_desert = yes
		is_planet_class_pc_tropical = yes
		is_planet_class_pc_arid = yes
		is_planet_class_pc_continental = yes
		is_planet_class_pc_ocean = yes
		is_planet_class_pc_tundra = yes
		is_planet_class_pc_arctic = yes
		is_planet_class_pc_alpine = yes
		is_planet_class_pc_savannah = yes
		is_planet_class_pc_nuked = yes
		is_planet_class_pc_gaia = yes
		AND = {
			alpha_mod_enabled = yes
			alpha_mod_habitable_planet = yes
		}	
		AND = { 
			new_worlds_enabled = yes
			new_worlds_habitable_planet = yes
		}	
		AND = {
			planetary_diversity_enabled = yes
			planetary_diversity_habitable_planet = yes
		}	
	}
}

is_gas_giant = {
	OR = {
		is_planet_class_pc_gas_giant = yes
		AND = { 
			new_worlds_enabled = yes
			is_planet_class_pc_gas_giant_hot = yes
			is_planet_class_pc_ice_giant = yes
		}	
		AND = {
			alpha_mod_enabled = yes
			is_planet_class_pc_gas_giant_habitable = yes 
		}	
	}
}

    OR = {
        solar_system = { any_planet = { is_planet_class_pc_m_star = yes } }
        AND = {
            real_space_enabled = yes
            OR = {
                solar_system = { any_planet = { is_planet_class_pc_l_star = yes } }
                solar_system = { any_planet = { is_planet_class_pc_d_star = yes } }
            }    
        }    
    }
}
