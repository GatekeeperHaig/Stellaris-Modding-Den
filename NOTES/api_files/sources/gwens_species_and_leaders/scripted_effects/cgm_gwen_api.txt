check_pop_traits_gwen = {
	if = {
		limit = { is_robot_pop = no }
		if = {
			limit = { has_trait = Gwen_Adaptation_Amoeba_Born }
			prev = {
				change_variable = { which = food_mult_planet_pop value = 0.10 }
			}
		}
		if = {
			limit = { has_trait = Gwen_Adaptation_Electro_Biogensisis }
			prev = {
				change_variable = { which = food_mult_planet_pop value = -0.20 }
				change_variable = { which = energy_weight_planet_pop value = 0.20 }
			}
		}
		if = {
			limit = { has_trait = Gwen_Adaptation_Energy_Being }
			prev = {
				change_variable = { which = energy_weight_planet_pop value = -1.5 }
			}
		}
		if = {
			limit = { has_trait = Gwen_Adaptation_Herbivorous_Species }
			prev = {
				change_variable = { which = energy_mult_planet_pop value = -0.15 }
				change_variable = { which = minerals_mult_planet_pop value = -0.15 }
				change_variable = { which = food_mult_planet_pop value = 0.15 }
			}
		}
		if = {
			limit = { has_trait = Gwen_Adaptation_Rooted_Species }
			prev = {
				change_variable = { which = energy_mult_planet_pop value = -0.20 }
				change_variable = { which = food_mult_planet_pop value = 0.10 }
				change_variable = { which = minerals_mult_planet_pop value = 0.10 }
			}
		}
		if = {
			limit = { has_trait = Gwen_Adaptation_Sad_Socrates }
			prev = {
				change_variable = { which = physics_research_mult_planet_pop value = 0.9 }
				change_variable = { which = engineering_research_mult_planet_pop value = 0.9 }
				change_variable = { which = society_research_mult_planet_pop value = 0.9 }
			}
		}
		if = {
			limit = { has_trait = Gwen_Adaptation_Pollinating }
			prev = {
				change_variable = { which = food_weight_planet_pop value = 0.15 }
			}
		}
	}
	if = {
		limit = { is_robot_pop = yes }
		if = {
			limit = { has_trait = Gwen_Adaptation_organic_simulation }
			prev = {
				change_variable = { which = influence_weight_planet_pop value = 0.05 }
			}
		}
	}
}
