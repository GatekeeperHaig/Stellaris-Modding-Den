	id = cgm_pe_mtth_events.1
	hide_window = yes
	
	mean_time_to_happen = {	years = 2 }	
	
	trigger = { any_tile = { has_blocker = tb_lava_flows } }
	
	immediate = { 
		every_tile = {

					set_blocker = tb_catastrophe_aftermath_pyroclastic_terrain_1											
				}
				50 = { }
			}	
		}
	}	
}
