building_autochthon_monument = {
	#building defines
	tags = { unity_building }
	planet_unique = yes
	is_listed = yes
	base_buildtime = 180	
	cost = { minerals = 100 }	
	required_resources = { energy = 1 }	
	show_tech_unlock_if = { standard_unity_building_tech_unlock_conditions = yes }
	#requirements and upgrade
	potential = { conditions_standard_unity_1 = yes } 
	prerequisites = { "tech_cultural_heritage" }
	upgrades = { building_heritage_site }
	destroy_if = { standard_unity_building_destroy_conditions = yes }
	#resource output and modifiers		
	produced_resource_trigger = {
		modifier = { #this is the base building output using vanilla stats
			overhauled_building_content_enabled = no
			resources = { unity = 2 }
		}
		modifier = { #this is the base building output using overhauled stats
			overhauled_building_content_enabled = yes
			resources = { unity = 2 }
		}		
	}		
	#ai restrictions and weights	
	ai_allow = { tile_has_any_strategic_resource = no }		
	ai_weight = { 
		modifier = { 
			add = 50 
			check_variable = { which = "unity_building_tile" value = 1 }
		}
	}
}
