# Robot Armies
mem_super_robot_army = {
	has_species = no
	damage = 2.5
	health = 4.5
	morale = 2.0
	time = 90
	maintenance = 1.0
	icon_frame = 4
	cost = {
		minerals = 350
		energy = 50
	}
	
	potential = {
		custom_tooltip = {
			text = mem_super_robot_army_limit_tooltip
			planet = {
				any_tile = {
					has_building = building_mem_ancient_robot_building_2
				}
			}
		}
	}
	
	allow = {
		custom_tooltip = {
			text = mem_super_robot_army_limit_tooltip
			planet = {
				any_tile = {
					has_building = building_mem_ancient_robot_building_2
				}
			}
			owner = {
				NOT = { 
					 check_variable = {
						which = "mem_super_robot_army_count" 
						value = 4
					}
				}
			}
		}
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "mem_super_robot_army_count" 
				value = 1
			}
		}
	}
	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "mem_super_robot_army_count" 
				value = -1
			}
		}
	}
}
