@basicTypeCost = 2000

terraform_link = {
	from = "orbit_dusty_telluric"
	to = "orbit_wasteland"

	resources = {
		category = terraforming

		cost = {
			energy = @basicTypeCost
		}
	}

	duration = 1800

	potential = {
		from = { has_modifier = terraforming_candidate }
	}
	condition = {
		has_technology = "tech_atmospheric_manipulation"
	}

	effect = {
		from = { remove_uninhabitable_modifiers = yes }
		from = {
			add_modifier = {
				modifier = "lithosphere-uniform"
				days = -1
			}
			add_modifier = {
				modifier = "hydrosphere-water_limited"
				days = -1
			}
			add_modifier = {
				modifier = "biosphere-barren"
				days = -1
			}
			add_modifier = {
				modifier = "atmosphere-marginal"
				days = -1
			}
			spawn_gravity_modifier = yes
			check_for_terraforming_special_cases = yes
			spawn_planet_deposits = yes
		}
	}

	ai_weight = {
		weight = 5
	}
}
