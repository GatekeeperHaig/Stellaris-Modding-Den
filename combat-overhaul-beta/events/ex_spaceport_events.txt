namespace = ex_spaceport_events

planet_event = {
	id = ex_spaceport_events.2
	hide_window = yes

	trigger = {
		has_spaceport = yes		
		#is_colony = yes
		#has_owner = yes	
		is_spaceport_module_slot_free = 2
		has_spaceport_module = projectile_weapon
	}

	immediate = {
		remove_spaceport_module = { module = projectile_weapon }
		set_spaceport_module = { slot = 1 module = projectile_weapon_2 }
	}
}
