# To spawn special blockers after terraforming world with special modifier. Ensures the blockers are pressent. 
c_spawn_special_blockers = {

	if = { limit = { has_modifier = Android_Pleasure_Palace }
		random_tile = {
			limit = {
				AND = {
					has_building = no
					has_pop = no
					num_adjacent_tiles > 3
			}
		}
			limit = { has_blocker = yes }
				remove_blocker = yes
		}
			clear_deposits = yes
			set_blocker = "c_tb_Android_Pleasure_Palace"
		}
	}
	if = { limit = { has_modifier = Precursor_Planetary_Shield }
		random_tile = {
			limit = {
				AND = {
					has_building = no
					has_pop = no
					num_adjacent_tiles > 3
			}
		}
			limit = { has_blocker = yes }
				remove_blocker = yes
		}
			clear_deposits = yes
			set_blocker = "c_tb_Precursor_Planetary_Shield"
		}
	}
	if = { limit = { has_modifier = Temple_of_the_Ancient_One }
		random_tile = {
			limit = {
				AND = {
					has_building = no
					has_pop = no
					num_adjacent_tiles > 3
			}
		}
			limit = { has_blocker = yes }
				remove_blocker = yes
		}
			clear_deposits = yes
			set_blocker = "c_tb_Temple_of_the_Ancient_One"
		}
	}
}
# Colour planet name based on requirements and give it an icon
c_colour_planet_name = {
	if = { limit = {
			AND = {
				has_planet_flag = "pm_is_precursor_world"
				NOT = { has_planet_flag = "pm_is_precursor_world_colour" }
		} 
		set_name = "color_precursor"
		set_planet_flag = "pm_is_precursor_world_colour"
		log = "***- Planet given a blue colour and icon----"
		log = "***-------- Planet name: [This.GetName] ----"

	}
	if = { limit = {
			AND = {
				has_planet_flag = "pm_is_wondrous_world"
				NOT = { has_planet_flag = "pm_is_wondrous_world_colour" }
		} 
		set_name = "color_planet_wonder"
		set_planet_flag = "pm_is_wondrous_world_colour"
		log = "***- Planet given an orange colour and icon----"
		log = "***-------- Planet name: [This.GetName] ----"

	}
}

# Remove planet name and icon and reset name to something that makes sense
c_reset_planet_name = {
	if = { 
		limit = {
			OR = {
				has_planet_flag = "pm_is_precursor_world"
				has_planet_flag = "pm_is_wondrous_world"
				has_planet_flag = "pm_is_infested_world"
		}
		set_name = random
		set_name = "color_default_white"
		if = {	limit = {
				has_planet_flag = "pm_is_precursor_world"
		}
			# remove_planet_flag = "pm_is_precursor_world"   #never remove because these are always kept
			remove_planet_flag = "pm_is_precursor_world_colour"
		}
		if = {	limit = {
				has_planet_flag = "pm_is_wondrous_world"
		}
			remove_planet_flag = "pm_is_wondrous_world"
			remove_planet_flag = "pm_is_wondrous_world_colour"
		}
		if = {	limit = {
				has_planet_flag = "pm_is_infested_world"
		}
			remove_planet_flag = "pm_is_infested_world"
		}
	}	
}

		limit = { AND = { has_building = no	has_pop = no num_adjacent_tiles > 3 } }

		clear_deposits = yes

			16 = { set_blocker = "pm_tb_Gargantuan_Sinkhole" }
