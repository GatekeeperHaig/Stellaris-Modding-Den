namespace = pdcrystal

#EnterSystem
ship_event = {
	id = pdcrystal.10
	title = pdcrystal.10.name
	desc = pdcrystal.10.desc
	picture = GFX_evt_crystal
	location = from
	show_sound = event_ship_bridge

	is_triggered_only = yes

	trigger = {
		from = {
			has_star_flag = crystal_system
		}
		owner = {
			NOT = {
				has_country_flag = crystal_encounter
			}
			OR = {
				is_country_type = default
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
			}
		}
	}

	immediate = {
		owner = {
			set_country_flag = crystal_encounter
		}
	}

	option = {
		name = pdcrystal.10.a
	}
}

#Crsytal Temple
planet_event = {
	id = pdcrystal.30
	title = pdcrystal.30.name
	desc = pdcrystal.30.desc
	picture = GFX_evt_crystal3
	show_sound = event_activating_unknown_technology
	location = root

	trigger = {
		has_planet_flag = crystal_planet
		has_owner = yes
		original_owner = yes
		is_homeworld = no		
		owner = {
			NOR = {
				has_country_flag = colony_crystal_temple
				has_authority = auth_machine_intelligence
			}
		}
		has_ground_combat = no
		is_occupied_flag = no
		free_building_tiles > 0
	}

	is_triggered_only = yes

	immediate = {
		owner = { save_event_target_as = crystal_choice }
		set_planet_flag = colony_event		
		set_planet_flag = crystal_temple_choice
		owner = {
			set_country_flag = colony_crystal_temple
		}
		random_tile = {
			limit = {
				has_building = no
				has_grown_pop = no
				has_growing_pop = no
				has_blocker = no
			}
			set_building = building_crystaltemple
		}
	}

	option = {
		name = "ONSCREEN"
		hidden_effect = {
			owner = { country_event = { id = pdcrystal.40 } }
		}
	}
}

# Incoming Transmission

	id = pdcrystal.40
	title = "pdcrystal.40.name"
	desc = "pdcrystal.40.desc"
	
	diplomatic = yes
	
	is_triggered_only = yes
	
	immediate = {	
		random_owned_leader = {
			limit = {
				leader_class = scientist
			}
			save_event_target_as = scientist_portrait
			}
	}
		
	picture_event_data = {
		portrait = event_target:scientist_portrait
		room = temple_room
	}
	
	option = {
		name = pdcrystal.40.a
		response_text = pdcrystal.40.a.response
		is_dialog_only = yes
	}
	option = {
		name = pdcrystal.40.b
		response_text = pdcrystal.40.b.response
		is_dialog_only = yes
	}
	option = {
		name = pdcrystal.40.c
		response_text = pdcrystal.40.c.response
		is_dialog_only = yes
	}
	option = {
		name = pdcrystal.40.d
		response_text = pdcrystal.40.d.response	
		hidden_effect = {
			FROM = { 
				planet_event = { id = pdcrystal.50 days = 14 }
			}
		}
