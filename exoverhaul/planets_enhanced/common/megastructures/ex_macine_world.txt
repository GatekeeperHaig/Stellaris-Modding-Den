ex_cybertronic_world = {
	entity = ""
	construction_entity = "machine_planet"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = yes
	entity_offset = { x = 0 y = 0 }
	build_time = 2700
	build_cost = {
		energy = 10000
		minerals = 10000		
		influence = 150
	}
	construction_blocks_others = no

	potential = {	
		has_ascension_perk = ap_machine_worlds
		has_global_flag = ex_cybertronic_worlds_active
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}

	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_replace_molten_world"
				is_planet_class = pc_molten
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}			
		}
	}	
	
	ai_weight = {
		factor = 10000
	}
	
	on_build_start = {}
	on_build_cancel = {}
	
	on_build_complete = {
		fromfrom.planet = {
			remove_planet = yes
		}	
		spawn_planet = {
			class = "pc_ex_cybertronic"
			size = 25
			location = fromfrom.planet
			orbit_angle_offset = 0
			init_effect = {
				set_name = "Habitable_Artificial_World"
				set_planet_entity = {
					entity = "machine_planet"
					graphical_culture = fromfrom
				}
				surveyed = {
					set_surveyed = yes
					surveyor = FROM
				}
				set_all_comms_surveyed = yes
				while = {
					limit = {
						any_tile = { has_blocker = yes }
					}
					random_tile = {
						limit = { has_blocker = yes }
						remove_blocker = yes
					}
				}				
				trigger_megastructure_icon = yes
			}
		}
		remove_megastructure = fromfrom
	}	
}
