tb_mountain_range = {
	time = 200
	cost = {
		energy = 400
	}

	spawn_chance = {
	  modifier = {
			add = @buckets
			planet = {
				is_colonizable = yes
				has_modifier = "mountainous_world"
			}
		}
		modifier = {
	    add = @lot
	    planet = {
	      is_colonizable = yes
				OR = {
					is_planet_class = pc_alpine
					is_planet_class = pc_cascadian
					is_planet_class = pc_hajungle
				}
			}
	  }
		modifier = {
	    add = @enough
	    planet = {
	      is_colonizable = yes
				NOT = { is_planet_class = pc_floating }
				NOT = { is_planet_class = pc_city }
				NOT = { is_planet_class = pc_desertislands }
				NOT = { is_planet_class = pc_ocean }
				NOT = { is_planet_class = pc_alpine }
				NOT = { is_planet_class = pc_cascadian }
				NOT = { is_planet_class = pc_hajungle }
				NOT = { is_planet_class = pc_subarctic }
			}
	  }
		modifier = {
	    add = @little
	    planet = {
	      is_colonizable = yes
				OR = {
					is_planet_class = pc_desertislands
					is_planet_class = pc_ocean
					is_planet_class = pc_subarctic
				}
			}
	  }
	  modifier = {
	    factor = 0.25
	    planet = {
	      has_modifier = "titanic_life"
	    }
	  }
	  modifier = {
	      factor = 0.2
	      planet = {
	        has_modifier = "boundless_plains"
	      }
	  }
	  modifier = {
	    factor = 0.25
	    planet = {
	      planet_size < 15
	    }
	  }
	}

	prerequisites = { "tech_tb_mountain_range" }
}
