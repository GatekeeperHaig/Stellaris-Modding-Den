planetary_diversity_unique_planet_init = {
	clear_planet_modifiers = yes
	every_tile = { 
		limit = { has_blocker = yes }
		remove_blocker = yes
	}
	if = { 
		limit = { is_planet_class = pc_crystal }
		add_modifier = { modifier = crystal days = -1 }		
		while = { 
			count = 3 
			random_tile = {
				limit = { has_deposit = no }
				set_deposit = d_crystal_deposit
			}
		}		
		#example regular resource spawns
		#the effect itself - planetary_diversity_planetary_resource_spawn_table - is stored in the file pd_resource_spawn_table. Change the weights in that file to affect the odds of specific regular deposits appearing there.
		#this effect spawns regular deposits on tiles. Specify the relative amount of tiles you want to have deposits by changing the odds in the random list
		every_tile = { 
			limit = { has_deposit = no }
			random_list = {
				85 = { planetary_diversity_planetary_resource_spawn_table = yes }
				15 = { }
			}				
		}
		every_tile = {
			limit = { has_blocker = no has_building = no }
			random_list = { 
				30 = { planetary_diversity_tile_blocker_spawn_table = yes }
				70 = { }
			}
		}		
		#you can specify the amount of repetitions, hence number of tiles with this deposit like this
	}
	else_if = { 
		limit = { is_planet_class = pc_floating }
		add_modifier = { modifier = floating days = -1 }
		while = { 
			count = 3 
			random_tile = {
				limit = { has_deposit = no }
				set_deposit = d_eludium_deposit
			}
		}						
		every_tile = { 
			limit = { has_deposit = no }
			random_list = {
				85 = { planetary_diversity_planetary_resource_spawn_table = yes }
				15 = { }
			}				
		}
		every_tile = {
			limit = { has_blocker = no has_building = no }
			random_list = { 
				30 = { planetary_diversity_tile_blocker_spawn_table = yes }
				70 = { }
			}
		}				
	}
	else_if = { 
		limit = { is_planet_class = pc_graveyard }
		add_modifier = { modifier = graveyard days = -1 }	
		every_tile = { 
			limit = { has_deposit = no }
			random_list = {
				85 = { planetary_diversity_planetary_resource_spawn_table = yes }
				15 = { }
			}				
		}
		every_tile = {
			limit = { has_blocker = no has_building = no }
			random_list = { 
				30 = { planetary_diversity_tile_blocker_spawn_table = yes }
				70 = { }
			}
		}				
	}
	else_if = { 
		limit = { is_planet_class = pc_mushroom }
		add_modifier = { modifier = mushroom days = -1 }
		every_tile = { 
			limit = { has_deposit = no }
			random_list = {
				85 = { planetary_diversity_planetary_resource_spawn_table = yes }
				15 = { }
			}				
		}
		every_tile = {
			limit = { has_blocker = no has_building = no }
			random_list = { 
				30 = { planetary_diversity_tile_blocker_spawn_table = yes }
				70 = { }
			}
		}				
	}	
	else_if = { 
		limit = { is_planet_class = pc_city }
		add_modifier = { modifier = city days = -1 } 
		#example of a modifier added to a planet
		#you can do as many modifiers as you want like this. 
		every_tile = { 
			limit = { has_deposit = no }
			random_list = {
				85 = { planetary_diversity_planetary_resource_spawn_table = yes }
				15 = { }
			}				
		}				
	}
}
