namespace = ex_spaceport_events

event = { #this event fires on game start, firing ex_spaceport_events.10
	id = ex_spaceport_events.0
	hide_window = yes
	is_triggered_only = yes

	immediate = { 
		every_country = { 
			limit = { is_country_type = default }
			every_owned_planet = { 
				set_planet_flag = game_start_economic_spaceport		
				if = {
					limit = { 
						has_spaceport = yes
						is_spaceport_module_slot_free = 1
					}
					planet_event = { id = ex_spaceport_events.10 }
				}	
			}
		}
	}	
}
