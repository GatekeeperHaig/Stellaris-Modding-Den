# Solar Storms
event = {
	id = pm_solar.10
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		log = "---- Fired Event pm_solar.10 ----"
		random_system = {
			limit = { NOT = { has_star_flag = pm_solar_event } }
			random_system_planet = {
				limit = {
					AND = {
						is_star = yes
						pm_is_planet_star = yes
					}
				}
				add_modifier = { modifier = "Violent_Sun" days = 180 }
			}
			every_system_planet = {
				limit = { NOT = { is_star = yes	} }
				add_modifier = { modifier = "Solar_Storm" days = 180 }
			}
			set_timed_star_flag = { flag = pm_solar_event days = 360 }
		}
	}
}

event = {
	id = pm_solar.40
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		log = "---- Fired Event pm_solar.40 ----"

						has_modifier = Periodic_Meteor_Showers

					}
				}
			}

		}
	}
