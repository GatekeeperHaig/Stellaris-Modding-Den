check_planets_enhanced_sr_buildings = {
	if = {
		limit = { has_building_sr_ex_unique_5 = yes }
		every_neighboring_tile = { change_variable = { which = engineering_research_weight value = 3 } }
		break = yes
	}	
	if = {
		limit = { has_building_sr_ex_planetary_1 = yes }
		every_neighboring_tile = { change_variable = { which = minerals_weight value = 2 } }
		break = yes
	}		
	if = {
		limit = { has_building_sr_engos_2 = yes }
		every_neighboring_tile = { 
			change_variable = { which = minerals_weight value = 1 } 
			change_variable = { which = energy_weight value = 1 } 
		}	
		break = yes
	}			
}	

check_planets_enhanced_planet_modifiers = {
	if = {
		limit = { has_modifier = ex_tw } 
		every_tile = {
			change_variable = { which = society_research_weight value = 2 }
			change_variable = { which = physics_research_weight value = 2 }
			change_variable = { which = engineering_research_weight value = 2 }		
		}	
	}
	if = {
		limit = { has_modifier = ex_necropolis }
		every_tile = { change_variable = { which = minerals_weight value = 2 } }
	}
}	

check_planets_enhanced_planet_bonus_buildings = {
	if = {
		limit = { has_building = building_sr_pitharan_1 }
		every_tile = { change_variable = { which = food_weight value = 2 } }
}
