tech_unlock_conditions_military_academy = { is_machine_empire = no }
tech_unlock_conditions_slave_processing = {
	OR = {
		is_hive_mind_empire = yes
		AND = { has_policy_flag = slavery_allowed is_machine_empire = no }
	}
}

	planets_enhanced_enabled = no
	OR = {
		AND = { overhauled_building_stats_enabled = no is_machine_empire = no }
		AND = { overhauled_building_stats_enabled = yes is_gestalt_consciousness = no }
	}
}

	planets_enhanced_enabled = no
	OR = {
		AND = { overhauled_building_stats_enabled = no is_machine_empire = yes }
		AND = { overhauled_building_stats_enabled = yes is_gestalt_consciousness = yes }
	}
}
