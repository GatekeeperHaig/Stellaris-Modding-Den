spawn_temperature_zone_modifier = {
	if = { 			
		limit = { has_planet_flag = "torrid_zone" }
		remove_vanilla_planet_modifiers = yes
		add_modifier = {
			modifier = "temperature-scorching"
			days = -1
		}
	}

spawn_gravity_modifier = {
	random_list = {
		33 = {
			add_modifier = {
				modifier = "low_gravity"
				days = -1
			}
			modifier = { factor = 2.00 planet_size < 15 }
			modifier = { factor = 0.25 planet_size > 18 }
		}
		33 = {
			add_modifier = {
				modifier = "standard_gravity"
				days = -1
			}
			modifier = { factor = 2.00 AND = { planet_size > 14 planet_size < 24 } }
		}
		33 = {
			add_modifier = {
				modifier = "high_gravity"
				days = -1
			}
			modifier = { factor = 2.00 planet_size > 23 }
			modifier = { factor = 0.25 planet_size < 20 }
		}
	}
}
