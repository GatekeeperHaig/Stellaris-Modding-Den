country_edict = {
	name = "disable_ai_boost"
	potential = { 
		is _ai = no
		has_global_flag = ai_boost_enabled
	}
	
	effect = { hidden_effect = { remove_global_flag = ai_boost_enabled } }	
}

country_edict = {
	name = "enable_ai_boost"
	potential = { 
		is _ai = no
		NOT = { has_global_flag = ai_boost_enabled }
	}
	
	effect = { hidden_effect = { set_global_flag = ai_boost_enabled } }	
}
