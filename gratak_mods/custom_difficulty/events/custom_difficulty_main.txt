#game start init
country_event = {
  id = custom_difficulty.2
  is_triggered_only = yes
  hide_window = yes
  trigger = {
    is_ai = no
  }
  immediate = {
    if = {
      limit = {
        is_difficulty = 0
      }
      set_global_flag = custom_difficulty_ensign
      country_event = { id = custom_difficulty.11 }
    }
    if = {
      limit = {
        is_difficulty = 2
      }
      set_global_flag = custom_difficulty_captain
      country_event = { id = custom_difficulty.12 }
    }
    if = {
      limit = {
        is_difficulty = 3
      }
      set_global_flag = custom_difficulty_commodore
      country_event = { id = custom_difficulty.13 }
    }
    if = {
      limit = {
        is_difficulty = 4
      }
      set_global_flag = custom_difficulty_admiral
      country_event = { id = custom_difficulty.14 }
    }
    if = {
      limit = {
        is_difficulty = 5
      }
      set_global_flag = custom_difficulty_grand_admiral
      country_event = { id = custom_difficulty.15 }
    }
    if = {
      limit = {
        is_difficulty = 1
      }
      set_global_flag = custom_difficulty_scaling
      country_event = { id = custom_difficulty.16 }
    }
  }
}
