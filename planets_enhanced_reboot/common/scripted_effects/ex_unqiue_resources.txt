set_planetary_resource_sr_ex_unique_1 = {
	if = {
		limit = { 
			NOT = { every_system = { any_planet = { has_planet_flag = planetary_resource_sr_ex_unique_1_rolled } } }			
		}
		random_system = {
			limit = {
				has_owner = no
				any_planet = { conditions_resource_sr_ex_unique_1 = yes	}	
			}
			random_planet = { 
				limit = { conditions_resource_sr_ex_unique_1 = yes }
				set_planet_flag = planetary_resource_sr_ex_unique_1_set
				random_tile = { 
					if = { limit = { has_building = yes } remove_building = yes } 
					set_deposit = d_sr_ex_unique_1_deposit
				}	
			}				
		}			
	}	
}
