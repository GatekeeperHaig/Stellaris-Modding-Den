	id = gpm_debug.10000

	is_triggered_only = yes

	immediate = {

				gpm_crisis_purge_modifiers = yes

				gpm_spawn_fallen_empire_modifier = yes
				gpm_spawn_fallen_empire_modifier = yes

					50 = {}

			}
			if = {
				limit = {
					NOT = { is_capital = yes }

				gpm_crisis_purge_modifiers = yes
				gpm_remove_precursor_modifiers = yes

				gpm_spawn_fallen_empire_modifier = yes
				random_list = {
					50 = {}
					50 = { gpm_spawn_fallen_empire_modifier = yes }
				}
				# Chance to spawn one fallen empire modifier
				random_list = {
					50 = {}
					50 = { gpm_spawn_fallen_empire_modifier = yes }
				}
