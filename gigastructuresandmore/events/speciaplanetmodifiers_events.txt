namespace = planet_type

planet_event = {
	id = planet_type.1
	title = EVENT
	desc = EVENT
	hide_window = yes
	#is_triggered_only = yes
	fire_only_once = yes
	
	immediate = {
		every_planet = {
			limit = {
				is_planet_class = pc_ll_temperate
			}
			add_modifier = {
				modifier = lifeless_temperate
				days = -1
			}
		}
	}
}

planet_event = {
	id = planet_type.2
	hide_window = yes

	trigger = {
		is_planet_class = pc_virtual_world
		NOT = { has_planet_flag = modifier_active }	
	}
	
	immediate = {
		set_planet_flag = modifier_active
		set_planet_flag = colony_event
		add_modifier = {
			modifier = virtual_world
			days = -1
		}
	}
}
