# Subterranean blockers

d_rock = {	
	time = 270
	use_for_min_max_adjustments = no
	is_for_colonizeable = yes
	category = deposit_cat_blockers
	
	resources = {
		category = deposit_blockers
		cost = {
			energy = 500
		}
	}
	
	potential = {
		OR = {
			is_planet_class_pc_underground_rocky = yes
			is_planet_class_pc_underground_icy = yes
		}
	}
	
	planet_modifier = { 
		planet_max_districts_add = -1
	}
	
	on_cleared = {
		random_list = {
			50 = { 
				add_deposit = d_mineral_rich_rock
			}
			25 = {
				add_deposit = d_geothermal_rock
			}
			25 = {
				# nothing
			}
		}
		
	}
	
	drop_weight = {
		weight = @high
		modifier = {
			factor = 10
			is_planet_class_pc_underground_rocky = yes
		}
		modifier = {
			factor = 5
			is_planet_class_pc_underground_icy = yes
		}	
	}
	
	ai_weight = {
		weight = 10
	}
	
	#prerequisites = { "tech_mountain_range" }
}
