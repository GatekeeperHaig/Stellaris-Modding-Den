ap_galactic_ascendancy = {
	on_enabled = {
		
		custom_tooltip = "become_ascended_empire"
		custom_tooltip = "describe_ascended_empire"
		
		give_technology = { tech = tech_dark_matter_propulsion	message = no }
		
		hidden_effect = {
			country_event = { id = fallen_technologies.1 }
		}
	}
	
	potential = {
		NOR = {
			has_authority = "auth_machine_intelligence"
			has_authority = "auth_hive_mind"
		}
		NOT = {
			has_ascension_perk = ap_galactic_ascendancy
		}
		NOT = { 
			host_has_dlc = "Apocalypse"
			has_technology = "tech_ascension_theory_apoc"
		}
	}
	
	possible = {	
		custom_tooltip = {
			fail_text = "requires_technology_dark_matter_power"
			has_technology = tech_dark_matter_power_core
		}		
		custom_tooltip = {
			fail_text = "requires_technology_ascension_theory"
			has_technology = tech_ascension_theory
		}			
		custom_tooltip = {
			fail_text = "requires_technology_fallen_technologies_2"
			has_technology = tech_fallen_technologies_2
		}
		custom_tooltip = {
			fail_text = "is_ascended_empire_subject"
			NOT = {
				any_country = {
					is_country_type = ascended_empire
					is_overlord_to = root
				}
			}
		}
		custom_tooltip = {
			fail_text = "is_awakened_empire_subject"
			NOT = {
				any_country = {
					is_country_type = awakened_fallen_empire
					is_overlord_to = root
				}
			}
		}
		custom_tooltip = {
			fail_text = "is_another_empire_subject"
			NOT = {
				any_country = {
					NOR = {
						is_country_type = awakened_fallen_empire
						is_country_type = ascended_empire
					}
					is_overlord_to = root
				}
			}
		}
		
		OR = {
			has_ethic = ethic_fanatic_spiritualist
			has_ethic = ethic_fanatic_materialist
			has_ethic = ethic_fanatic_xenophile
			has_ethic = ethic_fanatic_xenophobe
			has_ethic = ethic_fanatic_militarist
			has_ethic = ethic_fanatic_pacifist
			has_ethic = ethic_fanatic_authoritarian
			has_ethic = ethic_fanatic_egalitarian
		}
		
	}
	
	modifier = {
		damage_vs_country_type_default_mult = 0.33
		country_war_exhaustion_mult = -0.33
	}
	
	ai_weight = {
		factor = 1000
		modifier = {
			factor = 0
			num_owned_planets < 10
		}
		modifier = {
			factor = 0
			NOT = { is_country_type = default }
		}
	}
}
