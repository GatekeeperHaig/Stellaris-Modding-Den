namespace = nw_anomaly

### Subterranean Candidates

ship_event = {
	id = nw_anomaly.1
	title = "nw_anomaly.1.name"
	desc = {
		text = nw_anomaly.1.desc.1
		trigger = { 
			NOT = { 
				owner = { has_technology = "tech_terrestrial_sculpting" } 
			}
		}
	}
	desc = {
		text = nw_anomaly.1.desc.2
		trigger = { 
			owner = { has_technology = "tech_terrestrial_sculpting" }
		}
	}
	picture = GFX_evt_cold_barren_planet
	show_sound = event_scanner
	location = FROM
	
	is_triggered_only = yes
	
	option = {
		name = FASCINATING
		custom_tooltip = TERRAFORMING_CANDIDATE_EFFECT
		hidden_effect = {
			from = {
				add_modifier = {
					modifier = "subterranean_rocky_candidate"
					days = -1
				}
			}		
		}
	}
}
