		limit = {
			NAND = {
				exists = owner
				owner = { has_authority = auth_machine_intelligence }
			}
		}

			limit = { has_building = building_cofh_se_xeno_food }

	}
	if = {
		limit = {
			planet = {
				any_pop = {
					OR = { has_modifier = pop_enslaved_livestock has_modifier = pop_purging_processing }
				}
			}
		}

			limit = { has_building = building_cofh_se_xeno_food }

		}
	}
}
