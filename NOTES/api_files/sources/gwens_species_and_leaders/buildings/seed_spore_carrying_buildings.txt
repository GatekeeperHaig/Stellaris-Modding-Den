	base_buildtime = 180
	empire_unique = yes
	
	cost = {
		food = 200
		influence = 100
	}
	
	adjacency_bonus = {
		tile_building_resource_minerals_add = 0.75
		tile_building_resource_food_add = 0.75
		tile_building_resource_energy_add = 0.75
		tile_building_resource_unity_add = 0.75
		tile_building_resource_society_research_add = 0.75
		tile_building_resource_physics_research_add = 0.75
		tile_building_resource_engineering_research_add = 0.75
	}
	
	show_tech_unlock_if = { NOT = { has_authority = auth_machine_intelligence } }

	potential = {
		NOT = {
			owner = { has_authority = auth_machine_intelligence }
		}
		planet = {
			NOR = {
				is_planet_class = pc_habitat
				is_planet_class = pc_ringworld
			}
		}
	}

	destroy_if = {
		OR = {
			AND = {
				exists = owner
				owner = { has_authority = auth_machine_intelligence }
			}
		}
	}	
	
	country_modifier = {
#		
	}
	
	upgrades = {
		building_gwen_the_great_growth_t2
	}
