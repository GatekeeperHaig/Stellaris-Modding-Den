# Completely Leveled						-- overwriting to stop clearing planet wonder blockers
ship_event = {
	id = anomaly.205
	title = "anomaly.205.name"
	desc = "anomaly.205.desc"
	show_sound = event_structural_collapse
	picture = GFX_evt_ship_in_orbit
	location = FROM

	is_triggered_only = yes

	immediate = {
		from = {
			while = {
				limit = {
					any_tile = { 
						AND = {
							has_blocker = yes 
							NOT = {									# -guilli
								gpm_has_gpm_special_blocker = yes
							}
						}
					}
				}
				random_tile = {
					limit = { 
						AND = {
							has_blocker = yes
							NOT = {									# -guilli
								gpm_has_gpm_special_blocker = yes
							}
						}					
					}
					remove_blocker = yes
				}
			}
		}
	}

	option = {
		name = anomaly.205.a
		custom_tooltip = "anomaly.205.a.tooltip"
	}
}

# Grimacing - special project followup
ship_event = {
	id = anomaly.233
	title = anomaly.233.name
	desc = anomaly.233.desc
	picture = GFX_evt_physics_research
	show_sound = event_mystic_reveal
	location = fromfrom

	is_triggered_only = yes

	immediate = {
		fromfrom = {
			change_pc = pc_arid
			
			if = {									# -guilli
				limit = {
					NOR = {
						gpm_has_precursor_modifiers = yes
						has_modifier = "Planet_Wonder_Discovered"
					}
				}
				reset_planet = yes
				gpm_remove_precursor_modifiers = yes 	# -guilli
			}
		}
	}
	option = {
		name = MARVELOUS
		owner = {
			add_monthly_resource_mult = {
				resource = physics_research
				value = @tier3researchreward
				min = @tier3researchmin
				max = @tier3researchmax
			}
			add_monthly_resource_mult = {
				resource = engineering_research
				value = @tier3researchreward
				min = @tier3researchmin
				max = @tier3researchmax
			}
		}
	}
}
