pd_tr_biosynth_adopt = { # Access to LM Buildings
	possible = {
		has_ascension_perk = ap_fours_children
	}
	custom_tooltup = pd_tr_biosynth_adopt_desc # Gives to tech_biosynth_lm_refinery
	on_enabled = {
		hidden_effect = {
			country_event = { id = pdtechnoorganic.202 }
		}
	}
}

		}
	}

		ascension_perks_add = 1
	}
}
