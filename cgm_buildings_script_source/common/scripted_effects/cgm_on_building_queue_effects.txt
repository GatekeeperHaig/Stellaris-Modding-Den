### Effects stored in this file are fired from the on_building_queue/unqueue effect fields in the building files.

check_artist_monuments = {
	planet.owner = {
		if = {
			limit = { check_variable = { which = artist_monuments_built value < 1 } }
			set_variable = { which = artist_monuments_built value = 1 }
			else = { change_variable = { which = artist_monuments_built value = 1 }	}
		}
	}
}

subtract_artist_monuments = {
	planet.owner = { 
		change_variable = { which = artist_monuments_built value = -1 }
	}
}
