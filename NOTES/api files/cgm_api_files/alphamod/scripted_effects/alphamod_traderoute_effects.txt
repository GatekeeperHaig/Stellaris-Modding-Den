	add_modifier = {
		modifier = export_route_1
		days = -1 
	}
	set_planet_flag = pf_me
	owner = {
		random_owned_planet = {
			limit = {
				has_trade_building = yes
				not = { has_modifier = import_route_1 }	
				not = { has_planet_flag = pf_me }
			}
			add_modifier = {
				modifier = import_route_1
				days = -1 
			}
		}
	}
	remove_planet_flag = pf_me
}
