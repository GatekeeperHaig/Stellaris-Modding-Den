building_habitat_1 = {
	base_buildtime = 240
	planet_unique = yes
	
	cost = { minerals = 150 }
	
	allow = {
		custom_tooltip = {
			text = "requirement_not_homeworld"
			planet = { is_homeworld = no }
		}		
	}
	
	potential = {
		owner = { NOT = { has_authority = auth_machine_intelligence } } 
		planet = {
			NOR = {
				is_planet_class = pc_gaia
				is_planet_class = pc_ringworld_habitable
				is_planet_class = pc_habitat
				is_planet_class = pc_machine
			}		
		}	
	}
	
	required_resources = {
		minerals = 2
		food = 1
	}
	
	destroy_if = {
		AND = {
			exists = owner
			owner = { has_authority = auth_machine_intelligence }
		}
	}		
	
	produced_resources = { energy = 1 }	
	planet_modifier = { pop_environment_tolerance = 0.10 }	
	upgrades = { building_habitat_2 }
	ai_allow = { tile = { OR = { has_deposit = no has_resource = { type = energy amount > 0 } } } }	
	ai_weight = { weight = 10 }	
}
