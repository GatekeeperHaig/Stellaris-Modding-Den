namespace = ex_tw_modifier_events

planet_event = {
	id = ex_tw_modifier_events.1
	hide_window = yes
	
	trigger = {
		root = {
			is_planet_class = pc_nuked
			NOT = { has_planet_flag = ex_tw_modifier_rolled }
		}	
	}	
	
	immediate = {
		root = {
			random_list = {
				40 = { add_modifier = { modifier = ex_tw } }		
				30 = { add_modifier = { modifier = ex_necropolis } }
				20 = { 
					add_modifier = { modifier = ex_tw }
					add_modifier = { modifier = ex_necropolis }
				}
			}	
		}							
		set_planet_flag = ex_tw_modifier_rolled											
	}
}
