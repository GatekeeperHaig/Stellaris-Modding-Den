	required_resources = { energy = 5 }
	planet_modifier = { planet_orbital_bombardment_damage = -0.05 }
	army_modifier = { army_health = 0.25 }
	produced_resources = {
		engineering_research = 2
		physics_research = 2

		planet = { NOT = { is_planet_class = pc_habitat } }
		planet = { owner = { has_technology = tech_planetary_shield_generator } }

			}

				planet = {
					OR = {
						has_building = "building_capital_1"

						has_building = "building_capital_3"
					}
				}
			}
		}
		if = {
			limit = { owner = { has_authority = auth_machine_intelligence } }
			custom_tooltip = {

						has_building = "building_machine_capital_1"

					}
				}
			}
		}
	}
	prerequisites = { "tech_scaled_one_scaled_designs" }
	ai_allow = {
		NOR = {
			has_resource = { type = sr_betharian amount > 0 }

		}
		fromfrom = { has_technology = tech_planetary_shield_generator }
		NOT = { owner = { has_country_flag = cgm_disable_vanilla_building_AI } }
	}
	ai_weight = { weight = 0 }
