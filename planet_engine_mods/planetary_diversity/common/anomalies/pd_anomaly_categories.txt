# Hothouse Terraforming Candidate
anomaly_category = {
	key = "hhterraforming_candidate_category"
	desc = "hhterraforming_candidate_category_desc"
	picture = GFX_evt_toxic_planet
	level = 1
	#risk = 0
	
	spawn_chance = {		
		modifier = {
			add = 2
			NOT = { has_modifier = terraforming_candidate }
			OR = {
				is_planet_class = pc_hothouse
			}	
			planet_size > 9
			planet_size < 26	
			solar_system = {
				NOR = {
					is_star_class = sc_black_hole
					is_star_class = sc_neutron_star
					is_star_class = sc_pulsar
				}
			}
		}
	}
	
	on_success = {
	}

	on_fail = {
	}

	on_critical_fail = {
	}
}
