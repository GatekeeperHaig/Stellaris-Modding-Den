namespace = ex_ai_events

country_event = {
	id = ex_ai_events.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_country_flag = resource_check_delay }
		has_global_flag = ai_boost_enabled
		is_ai = yes
		OR = {
			energy < 1
			minerals < 1
			AND = {
				NOT = { has_authority = auth_machine_intelligence }
				food < 1
			}
		}		
	}			
	
	immediate = {
		if = {
			limit = {
				NOT = { has_authority = auth_machine_intelligence }
				food < 1
			}
			add_modifier = { modifier = ex_food_starvation days = 90 } 
		}
		if = {
			limit = { energy < 1 }
			add_modifier = { modifier = ex_energy_starvation days = 90 } 
		}
		set_timed_country_flag = { flag = resource_check_delay days = 90 }
	}
}
