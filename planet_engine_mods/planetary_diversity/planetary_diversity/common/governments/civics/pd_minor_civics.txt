#Hive World Start
civic_hive_world = {
	homeworld = pc_hive # Changes homeworld class regardless of what is picked for empire
	potential = { authority = { value = auth_hive_mind } }
	possible = {
		civics = {
			NOR = {
				value = civic_hive_floating_world
				value = civic_hive_graveyard_world
				value = civic_hive_technoorganic_world
				value = civic_hive_crystal_world
			}
		}
	}
	modification = no
	random_weight = {
		base = 0 #Should Not Happen
		modifier = {
			# civic has no effect after galaxy generation, disable for random generation
			factor = 0
			has_global_flag = game_started
		}
	}	
	description = "civic_tooltip_hive_world_effects"
}
