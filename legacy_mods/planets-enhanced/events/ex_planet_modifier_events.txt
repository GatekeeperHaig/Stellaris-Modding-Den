namespace = ex_planet_enhanced_events

planet_event = {
	id = ex_planet_enhanced_events.1
	hide_window = yes
	
	trigger = {
		OR = {
			is_planet_class = pc_ringworld_habitable
			is_planet_class = pc_nuked
			is_planet_class = pc_gaia
		}	
		NOT = { has_planet_flag = modifier_added }	
	}	
	
	immediate = { 
		set_planet_flag = modifier_added
		if = {
			limit = { is_planet_class = pc_ringworld_habitable }			
			add_modifier = { modifier = "ex_ring_world" days = -1 } 													
		}	
		if = {
			limit = { is_planet_class = pc_gaia }			
			add_modifier = { modifier = "ex_gaia_world" days = -1 } 														
		}			
		if = {
			limit = { is_planet_class = pc_nuked }
			random_list = {
				40 = { add_modifier = { modifier = ex_tw } }		
				30 = { add_modifier = { modifier = ex_necropolis } }
				20 = { 
					add_modifier = { modifier = ex_tw }
					add_modifier = { modifier = ex_necropolis }
				}
			}	
		}
	}		
}
