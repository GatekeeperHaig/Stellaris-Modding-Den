			random_list = {

					#change_pc = pc_primordial
					#set_entity_primordial = yes

					modifier = { add = 20 orbits_hot_star = yes }  #Hot stars better for Hot & Dry worlds
					modifier = { add = 0 orbits_collapsed_star = yes }
					modifier = { factor = 0.50 has_moon = yes }  #Having moons is good for more stable and temperate climates
					modifier = { factor = 1.50 has_moon = no }  #Without moons the likelihood of an extreme climate increases
				}
