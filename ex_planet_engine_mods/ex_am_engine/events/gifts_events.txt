namespace = giftsonature

planet_event = {
	id = giftsonature.2
	hide_window = yes

	is_triggered_only = false
	
	trigger = {
		has_building = building_warrior_house
		has_owner = yes
		not = { has_planet_flag = pf_ownerisatwar }
		owner = {	
			is_at_war = yes
		}
	}
	
	immediate = {
		set_planet_flag = pf_ownerisatwar
	}
}

planet_event = {
	id = giftsonature.3
	hide_window = yes

	is_triggered_only = false
	
	trigger = {
		has_planet_flag = pf_ownerisatwar
		has_owner = yes
		owner = {	
			is_at_war = no
		}
	}
	
	immediate = {
		remove_planet_flag = pf_ownerisatwar
	}
}
