@avoid = 0 
@standard = 1
@important = 10
@strategic = 100
@crucial = 1000

@reduce_interest_1 = 0.5
@reduce_interest_2 = 0.25
@reduce_interest_3 = 0.1

@increase_interest_1 = 2
@increase_interest_2 = 4
@increase_interest_3 = 10

ship_section_template = {
	key = "CORVETTE_MID_M1S1"
	ship_size = corvette
	fits_on_slot = mid
	entity = "corvette_M1S1_entity"
	icon = "GFX_ship_part_core_mid"
	
	prerequisites = { tech_torpedoes_1 tech_torpedoes_2 tech_torpedoes_3 tech_energy_torpedoes_1 tech_energy_torpedoes_2 tech_energy_torpedoes_3 }
	
	component_slot = {
		name = "SMALL_GUN_01"
		slot_size = small
		slot_type = weapon
		locatorname = "small_gun_01"
	}
	
	component_slot = {
		name = "TORPEDO_01"
		slot_size = torpedo
		slot_type = weapon
		locatorname = "medium_gun_01"
	}
	
	small_utility_slots = 2
	medium_utility_slots = 1
	aux_utility_slots = 1
	
	ai_weight = {
		weight = @standard
		modifier = {
			factor = @avoid
			NOT = { 
				any_country = { 
					AND = {
						is_enemy = yes
						has_technology = tech_spaceport_5
					}
				}	
			}								
		}
		modifier = {
			factor = @important
			any_country = { 
				AND = {
					is_enemy = yes
					any_owned_ship = {
						OR = {
							is_cruiser = yes
							is_battleship = yes
							is_dreadnought = yes
						}	
					}
				}								
			}
		}			
		modifier = {
			factor = @increase_interest_1
			any_country = { 
				AND = {
					is_enemy = yes
					any_owned_ship = { is_cruiser = yes }
				}
			}
		}	
		modifier = {
			factor = @increase_interest_2
			any_country = { 
				AND = {
					is_enemy = yes
					any_owned_ship = { is_battleship = yes }
				}
			}
		}	
		modifier = {
			factor = @increase_interest_3
			any_country = { 
				AND = {
					is_enemy = yes
					any_owned_ship = { is_dreadnought = yes }
				}
			}
		}
		modifier = {
			factor = @increase_interest_1
			OR = { 
				has_technology = tech_torpedoes_2			
				has_technology = tech_energy_torpedoes_2
			}
		}
		modifier = {
			factor = @increase_interest_2
			OR = { 
				has_technology = tech_torpedoes_3			
				has_technology = tech_energy_torpedoes_3
			}
		}		
	}	
}
