5minerals_building_planet_modifiers = { 5minerals_buildings_triggered = yes }
5minerals_adjacency = {
	every_neighboring_tile = {
		if = {
			limit = { has_building = building_mining_complex_1_5 }
			prevprev = {
				change_variable = { which = minerals_weight value = @tile_restricted_t1_adjacency_bonus }
			}
			else = {
				if = {
					limit = { has_building = building_mining_complex_2_5 }
					prevprev = {
						change_variable = { which = minerals_weight value = @tile_restricted_t2_adjacency_bonus }
					}
					else = {
						if = {
							limit = { has_building = building_mining_complex_3_5 }
							prevprev = {
								change_variable = { which = minerals_weight value = @tile_restricted_t3_adjacency_bonus }
							}
							else = {
								if = {
									limit = { has_building = building_mining_complex_4_5 }
									prevprev = {
										change_variable = { which = minerals_weight value = @tile_restricted_t4_adjacency_bonus }
									}
									else = {
										if = {
											limit = { has_building = building_mining_complex_5_5 }
											prevprev = {
												change_variable = { which = minerals_weight value = @tile_restricted_t5_adjacency_bonus }
											}
											else = { }
										}
									}
								}
							}
						}
					}
				}
			}
		}
	}
}
5minerals_buildings_triggered = {
	if = {
		limit = { has_building = building_mining_complex_5_5 }
		prev = {
			if = {
				limit = {
					planet = { has_capital_3 = yes }
				}
			}
			change_variable = { which = minerals_mult_planet_base value = @tile_restricted_t5_triggered_planet_modifier }
		}
		else = { }
	}
}
