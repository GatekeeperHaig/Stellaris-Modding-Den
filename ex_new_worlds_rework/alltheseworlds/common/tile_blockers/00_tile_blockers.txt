			}
		}
		modifier = {
			add = 20
			planet = {
				is_planet_class = pc_alpine
			}
		}
		modifier = {
			add = 13.5
			planet = {
				is_colonizable = yes
				OR = {
					is_planet_class = pc_continental
					is_planet_class = pc_retinal
