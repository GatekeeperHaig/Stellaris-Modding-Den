owner_allows_robots = { 
	exists = owner
	owner = { has_policy_flag = robots_allowed }
}

planet_has_robots = { 
	planet = { any_owned_pop = { is_robot_pop = yes } }
}	

is_robot_using_ai = { 
	is_ai = yes
	OR = {							
		has_ai_personality_behaviour = robot_exploiter
		has_ai_personality_behaviour = robot_liberator	
		is_machine_empire = yes
		is_synth_empire = yes
	}	
}	

owner_is_robot_using_ai = { 
	exists = owner
	owner = { 
		is_ai = yes
		OR = {							
			has_ai_personality_behaviour = robot_exploiter
			has_ai_personality_behaviour = robot_liberator	
			is_machine_empire = yes
			is_synth_empire = yes
		}	
	}	
}
