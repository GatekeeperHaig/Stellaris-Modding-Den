namespace = snowball_earth

#PD Snowball Earth Start
planet_event = {
	id = snowball_earth.01
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			owner = { has_origin = origin_pd_snowball_earth }
		}
	}

	immediate = {
		if = {
			limit = { has_planet_flag = pd_snowball_start_earth }
			pdsnowball_earth_system_setup = yes
			planet_event = { id = snowball_earth.02 days = 1 }
		}
	}
}

planet_event = {
	id = snowball_earth.02
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { has_planet_flag = "pd_snowball_start_earth" }
			clear_deposits = yes
			while = {
				count = 6
				add_deposit = "d_frozen_wasteland"
			}
			add_deposit = "d_massive_glacier"
			add_deposit = "d_massive_glacier"
			add_deposit = "d_hot_springs"
			add_deposit = "d_geothermal_vent"
			add_deposit = "d_ore_rich_caverns"
			add_deposit = "d_rich_mountain"
			add_deposit = "d_submerged_ore_veins"
			add_deposit = "d_forgiving_tundra"
			add_deposit = "d_forgiving_tundra"
			add_deposit = "d_fungal_caves"
		}
	}
}
