namespace = exo_alpha_mod_spawn_engine

event = { #this event triggers on game start
	id = exo_alpha_mod_spawn_engine.0
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {		
		set_global_flag = am_game_start
		every_country = {
			every_planet = {
				every_owned_pop = {
					limit = { is_robot_pop = no }
					ex_alpha_mod_replace_planet_pref_traits = yes
				}
			}
		}		
	}
}
