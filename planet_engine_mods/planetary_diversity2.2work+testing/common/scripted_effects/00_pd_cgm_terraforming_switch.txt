cgm_terraforming_switch = {
	from = {
		remove_negative_planet_modifiers = yes
		remove_modifier = terraforming_candidate
		if = {
			limit = { NOT = { has_modifier = terraformed } }
			add_modifier = { modifier = "terraformed" days = -1 }
		}
		if = { 
            limit = { is_planet_class = pc_city }
            every_tile = { 
                limit = { has_resource = { type = food amount > 0 } }
                clear_deposits = yes
			} 
			if = {
				limit = { NOT = { has_modifier = city } }
				add_modifier = { modifier = "city" days = -1 }
			}
        }    		
		every_tile = { 
			limit = { has_deposit = no }
			random_list = { 
				75 = { planetary_diversity_planetary_resource_spawn_table = yes }
				25 = { }
			}
		}	
	}
}
