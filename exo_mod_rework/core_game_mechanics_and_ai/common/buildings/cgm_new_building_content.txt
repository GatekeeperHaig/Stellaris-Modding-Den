		}

	#building defines
	icon = "building_uplink_node"
	tags = { unity_resource_buildings }
	planet_unique = yes
	is_listed = yes
	base_buildtime = 180	
	cost = { minerals = 100 }	
	required_resources = { energy = 1 }	
	show_tech_unlock_if = { 
		new_building_content_enabled = yes
		is_hive_mind_empire = yes
	}
	destroy_if = { 
		OR = {
			owner_is_hive_mind_empire = no
			new_building_content_enabled = no
		}	
	}
	#requirements and upgrade

		conditions_hive_unity_1 = yes
		new_building_content_enabled = yes

	on_unqueued = { subtract_unity_buildings = yes }		
	#ai restrictions and weights	
	ai_allow = { unity_buildings_ai_allow = yes }		
	ai_weight = { weight = 50 }
}
