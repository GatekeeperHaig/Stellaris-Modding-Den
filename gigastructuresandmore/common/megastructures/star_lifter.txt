	maintenance = {

	}
	
	monthly_production = {

	}
	
	potential = {
		AND = {
			has_ascension_perk = ap_galactic_wonders
			has_technology = tech_star_lifters

		every_system_planet = {
			limit = {

			change_pc = pc_frozen
		}
		every_system_planet = {
			limit = {
				is_planet_class = pc_barren
			}
			change_pc = pc_barren_cold
		}		
	}
