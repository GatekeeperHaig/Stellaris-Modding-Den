district_pdexotic_generator = {
	base_buildtime = 240

	min_for_deposits_on_planet = 3
	max_for_deposits_on_planet = 15

	show_on_uncolonized = {
		uses_district_set = pdexotic_standard
		FROM = { NOT = { has_origin = origin_pd_sulfur_springs } }
	}

	potential = {
		uses_district_set = pdexotic_standard
		OR = {
			NOT = { exists = owner }
			NOT = { owner = { has_origin = origin_pd_sulfur_springs } }
			AND = {
				owner = { has_origin = origin_pd_sulfur_springs }
				NOT = { is_planet_class = pc_sulfur }
			}
		}
		NOT = {
			AND = {
				owner = { is_ai = yes }
				owner = { country_uses_food = no }
			}
		}
	}

	allow = {
		NOT = { has_modifier = resort_colony }
	}

	conversion_ratio = 1
	convert_to = {
		district_generator_uncapped
		district_pdexotic_generator_uncapped
		district_generator
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = @base_cost
		}
		upkeep = {
			energy = 1
		}
	}

	planet_modifier = {
		planet_housing_add = 2
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_valid_civic = civic_agrarian_idyll
			}
		}
		modifier = {
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_technician_drone_add = 3
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			job_technician_drone_add = 2
			planet_housing_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		modifier = {
			job_technician_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		modifier = {
			job_fe_acolyte_generator_add = 2
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_technician_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		text = job_technician_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		text = job_fe_acolyte_generator_effect_desc
	}

	prerequisites = {
		tech_power_plant_1
	}

	ai_resource_production = {
		energy = 1
		trigger = {
			always = yes
		}
	}
}
