d_arid_highlands = {
	is_for_colonizeable = yes
	use_for_min_max_adjustments = yes
	
	use_weights_for_terraforming_swap_types = yes
	should_swap_deposit_on_terraforming = yes
	terraforming_swap_types = {
		d_hot_springs
		d_buzzing_plains # gaia
		d_strong_currents
	}
	
	category = deposit_cat_energy
	
	planet_modifier = { 
		district_generator_max = 1
	}
	
	potential = {
		OR = {
		is_dry = yes
		is_planet_class = pc_tidallylocked
		is_planet_class = pc_graveyard	
		is_planet_class = pc_ash	
		
		is_planet_class_pc_irradiated_marginal = yes
		is_planet_class_pc_hothouse = yes
		is_planet_class_pc_desolate = yes
		is_planet_class_pc_primordial = yes
		}
	}
	
	drop_weight = {
		weight = @high	
		modifier = {
			factor = @planet_type_bonus
			is_dry = yes
		}
	}
}
