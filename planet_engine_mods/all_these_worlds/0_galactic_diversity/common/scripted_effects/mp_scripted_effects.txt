mp_destroy_hyperlane = {
	random_neighbor_system = {
		limit = {
			NOR = {
				has_star_flag = "advanced_empire_space"
				has_star_flag = "special_empire_space"
				has_star_flag = "marauder_system"
			}
		}
		#ignore_hyperlanes = no
		remove_hyperlane = { 
			from = PREV
			to = THIS 
		}
	}
}

mp_disrupt_hyperlanes = {
	every_neighbor_system = {
		limit = {
			NOR = {
				#has_star_flag = "advanced_empire_space"
				#has_star_flag = "special_empire_space"
				#has_star_flag = "marauder_system"
			}
		}
		#ignore_hyperlanes = no
		remove_hyperlane = { 
			from = PREV
			to = THIS 
		}
	}
}

mp_create_hyperlane = {
	if = {
		limit = {
			any_neighbor_system = { 
				ignore_hyperlanes = yes
				NOT = {
					has_hyperlane_to = PREV
				}
			}
		}
		
		random_neighbor_system = {
			ignore_hyperlanes = yes
			limit = {
				NOR = {
					has_hyperlane_to = PREV
					#has_star_flag = "spawned_hyperlanes"
				}
			}
			
			add_hyperlane = { 
				from = PREV
				to = THIS 
			}
		}
	}
}

reroll_system_deposits = {
	every_system_planet = {
		clear_deposits = yes
		nw_orbital_resource_spawn_table = yes
	}
}
