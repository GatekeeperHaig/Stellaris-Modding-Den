### Colony Goes Missing
planet_event = {
	id = pdanomaly.20
	title = pdanomaly.20.name
	desc = pdanomaly.20.desc
	picture = GFX_evt_surreal_visions
	show_sound = event_wind_ruins
	location = ROOT

	pre_triggers = {
		has_owner = yes
		is_homeworld = no
		original_owner = yes
		is_ai = no
		has_ground_combat = no
		is_capital = no
		is_occupied_flag = no
	}

	trigger = {
		owner = {
			NOT = { is_gestalt = yes }
		}
		num_pops > 3
		OR = {
			is_wet = yes
			is_cold = yes
			is_dry = yes
		}
		NOR = {
			has_global_flag = pd_missing_colonists
			has_planet_flag = colony_event
		}
	}

	is_triggered_only = yes

	immediate = {
		set_global_flag = pd_missing_colonists
		set_planet_flag = colony_event
		#add_modifier = { modifier = "migrating_forests" days = -1 }
		random_owned_pop = {
			kill_pop = yes
		}
		random_owned_pop = {
			kill_pop = yes
		}
		random_owned_pop = {
			kill_pop = yes
		}
	}

	option = {
		name = pdanomaly.20.a
		enable_special_project = {
			name = "PD_MISSING_COLONISTS_PROJECT"
			location = this
			owner = root
		}
	}
}

	location = ROOT
	desc = {
		text = pdanomaly.22.desc.1
		trigger = { 
			NOT = { 
				owner = { has_technology = "tech_epigenetic_triggers" } 
			}
		}
	}
	desc = {
		text = pdanomaly.22.desc.2
		trigger = { 
			owner = { has_technology = "tech_epigenetic_triggers" }
		}
	}

	is_triggered_only = yes

	immediate = {
		owner = { 
			create_species = {
				name = random
				class = random
				portrait = pre_mam6
				traits = {
					ideal_planet_class = this
					trait = trait_presapient_docile_livestock
				}
				homeworld = FROM
				sapient = no
			}
		}
		create_pop = {
			species = last_created_species
		}
		create_pop = {
			species = last_created_species
		}
		create_pop = {
			species = last_created_species
		}
	}

	option = {
		name = pdanomaly.22.a
		owner = {
			IF = {
				limit = { owner = { has_technology = tech_epigenetic_triggers } }
				add_monthly_resource_mult = {
					resource = society_research
					value = 48
					min = 500
					max = 1500
				}
			}
			ELSE_IF = {
				limit = { owner = { NOT = { has_technology = tech_epigenetic_triggers } } }
			}
		}
	}
}
