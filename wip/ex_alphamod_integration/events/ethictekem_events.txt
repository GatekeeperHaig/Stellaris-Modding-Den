namespace = ethictekemevents

country_event = {
	id = ethictekemevents.1
	hide_window = yes

	is_triggered_only = false
	
	trigger = {
		has_country_flag = cf_wealthy_5
		has_country_strategic_resource = {
			type = sr_wealth
			amount < 5
		}
	}
	
	immediate = {
		remove_country_flag = cf_wealthy_5
		remove_modifier = wealthy_5
		remove_modifier = wealthy_10
		remove_modifier = wealthy_25
		remove_modifier = wealthy_50
		remove_modifier = wealthy_100
		remove_country_flag = cf_wealthy_10
		remove_country_flag = cf_wealthy_25
		remove_country_flag = cf_wealthy_50
		remove_country_flag = cf_wealthy_100
	}
}

planet_event = {
	id = ethictekemevents.2
	hide_window = yes

	is_triggered_only = false
	
	trigger = {
		has_modifier = satellite_network
		or = {
			has_spaceport = no
			nor = {
				has_building = building_rocket_launchsite
				has_building = building_shuttle_port
				has_building = building_advanced_shuttleport			
			}
		}		
	}
	
	immediate = {
		remove_modifier = satellite_network
	}
}	

country_event = {
	id = ethictekemevents.5
	hide_window = yes

	is_triggered_only = false
	
	trigger = {
		has_country_strategic_resource = {
			type = sr_wealth
			amount > 4
		}
		not = { has_country_flag = cf_wealthy_5 }
		has_country_strategic_resource = {
			type = sr_wealth
			amount < 10
		}
	}
	
	immediate = {
		set_country_flag = cf_wealthy_5
		add_modifier = {
			modifier = wealthy_5
			days = -1
		}
		remove_modifier = wealthy_10
		remove_modifier = wealthy_25
		remove_modifier = wealthy_50
		remove_modifier = wealthy_100
		remove_country_flag = cf_wealthy_10
		remove_country_flag = cf_wealthy_25
		remove_country_flag = cf_wealthy_50
		remove_country_flag = cf_wealthy_100
	}
}
