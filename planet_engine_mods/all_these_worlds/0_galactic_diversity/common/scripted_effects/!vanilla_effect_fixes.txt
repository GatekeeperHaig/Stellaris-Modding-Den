# Change planet class one step up in temperature
increase_planet_temperature = {
	# Clear Atmosphere Modifier
	if = {
		limit = { has_modifier = atmosphere-airless }
		remove_modifier = atmosphere-airless
	}
	if = {
		limit = { has_modifier = atmosphere-thin }
		remove_modifier = atmosphere-thin
	}
	if = {
		limit = { has_modifier = atmosphere-greenhouse }
		remove_modifier = atmosphere-greenhouse
	}
	
	# Torrid Zone
	if = {
		limit = { has_planet_flag = "torrid_zone" }
		if = {
			limit = { is_planet_class = "orbit_volcanic_telluric" }
			change_pc = orbit_molten
			add_modifier = {
				modifier = "atmosphere-thin"
				days = -1
			}
			spawn_uninhabitable_features = yes
		}
		else_if = {
			limit = { is_planet_class = "orbit_molten" }
			change_pc = orbit_iron_telluric
			add_modifier = {
				modifier = "atmosphere-airless"
				days = -1
			}
			spawn_uninhabitable_features = yes
		}
		else = {
			change_pc = orbit_volcanic_telluric
			add_modifier = {
				modifier = "atmosphere-thin"
				days = -1
			}
			spawn_uninhabitable_features = yes
		}
	}
	# Hot Zonoe
	if = {
		limit = { has_planet_flag = "hot_zone" }
		if = {
			limit = { is_planet_class = "orbit_volcanic_telluric" }
			change_pc = orbit_molten
			add_modifier = {
				modifier = "atmosphere-thin"
				days = -1
			}
			spawn_uninhabitable_features = yes
		}
		else_if = {
			limit = { is_planet_class = "orbit_molten" }
			change_pc = orbit_iron_telluric
			add_modifier = {
				modifier = "atmosphere-airless"
				days = -1
			}
			spawn_uninhabitable_features = yes
		}
		else = {
			change_pc = orbit_volcanic_telluric
			add_modifier = {
				modifier = "atmosphere-thin"
				days = -1
			}
			spawn_uninhabitable_features = yes
		}
	}
	# Warm Eco Zone
	if = {
		limit = { has_planet_flag = "warm_eco_zone" }
		if = {
			limit = { is_planet_class = "orbit_dusty_telluric" }
			random_list = {
				25 = {
					change_pc = pc_alpine
					convert_vanilla_habitable = yes
				}
				50 = {
					change_pc = orbit_volcanic_telluric
					add_modifier = {
						modifier = "atmosphere-thin"
						days = -1
					}
					spawn_uninhabitable_features = yes
				}
				25 = {
					# Stick with dusty
					add_modifier = {
						modifier = "atmosphere-thin"
						days = -1
					}
				}
				
			}
		}
		else_if = {
			limit = { is_planet_class = "orbit_volcanic_telluric" }
			random_list = {
				30 = {
					change_pc = pc_desert
					convert_vanilla_habitable = yes
				}
				20 = {
					change_pc = pc_alpine
					convert_vanilla_habitable = yes
				}
				50 = {
					# Stick with volcanic
					add_modifier = {
						modifier = "atmosphere-thin"
						days = -1
					}
				}				
			}
		}
		else = {
			change_pc = orbit_volcanic_telluric
			add_modifier = {
				modifier = "atmosphere-thin"
				days = -1
			}
			spawn_uninhabitable_features = yes
		}
	}
	# Temperate Eco Zone
	if = {
		limit = { has_planet_flag = "temperate_eco_zone" }
		if = {
			limit = { is_planet_class = "orbit_frozen_telluric" }
			random_list = {
				33 = {
					change_pc = pc_arctic
					convert_vanilla_habitable = yes
				}
				66 = {
					change_pc = orbit_water_panthalassic
					add_modifier = {
						modifier = "atmosphere-greenhouse"
						days = -1
					}
					spawn_uninhabitable_features = yes
				}				
			}
		}
		else_if = {
			limit = { is_planet_class = "orbit_water_panthalassic" }
			random_list = {
				30 = {
					change_pc = pc_ocean
					convert_vanilla_habitable = yes
				}
				20 = {
					change_pc = pc_continental
					convert_vanilla_habitable = yes
				}
				50 = {
					# Stick with panthalassic
					add_modifier = {
						modifier = "atmosphere-greenhouse"
						days = -1
					}
				}				
			}
		}
		else = {
			change_pc = orbit_frozen_telluric
			add_modifier = {
				modifier = "atmosphere-thin"
				days = -1
			}
			spawn_uninhabitable_features = yes
		}
	}
	# Cool Eco Zone
	if = {
		limit = { has_planet_flag = "cool_eco_zone" }
		if = {
			limit = { is_planet_class = "orbit_ammonia_panthalassic" }
			random_list = {
				50 = {
					change_pc = pc_arid
					convert_vanilla_habitable = yes
				}
				50 = {
					change_pc = orbit_dusty_telluric
					add_modifier = {
						modifier = "atmosphere-thin"
						days = -1
					}
					spawn_uninhabitable_features = yes
				}				
			}
		}
		else_if = {
			limit = { is_planet_class = "orbit_dusty_telluric" }
			random_list = {
				30 = {
					change_pc = pc_arctic
					convert_vanilla_habitable = yes
				}
				20 = {
					change_pc = pc_continental
					convert_vanilla_habitable = yes
				}
				50 = {
					# Stick with dusty
					add_modifier = {
						modifier = "atmosphere-thin"
						days = -1
					}
				}				
			}
		}
		else_if = {
			limit = { is_planet_class = "orbit_icy" }
			random_list = {
				30 = {
					change_pc = pc_arctic
					convert_vanilla_habitable = yes
				}
				20 = {
					change_pc = orbit_frozen_telluric
					add_modifier = {
						modifier = "atmosphere-thin"
						days = -1
					}
				}
				50 = {
					# Stick with icy
					add_modifier = {
						modifier = "atmosphere-airless"
						days = -1
					}
				}				
			}
		}
		else = {
			change_pc = orbit_ammonia_panthalassic
			add_modifier = {
				modifier = "atmosphere-greenhouse"
				days = -1
			}
			spawn_uninhabitable_features = yes
		}
	}
	# Cold Zone
	if = {
		limit = { has_planet_flag = "cold_zone" }
	}
	# Frigid Zone
	if = {
		limit = { has_planet_flag = "frigid_zone" }
	}
	
}

# Change planet class one step down in temperature

# Im unaware of any event that uses this effect, so I wont change it for now.
decrease_planet_temperature = {
	if = {
		limit = { is_planet_class = pc_molten }
		change_pc = pc_desert
		reroll_planet = yes
		clear_blockers = yes
	}
	else_if = {
		limit = { is_planet_class = pc_barren }
		change_pc = pc_arid
		reroll_planet = yes
		clear_blockers = yes
	}
	else_if = {
		limit = { is_planet_class = pc_desert }
		change_pc = pc_ocean
		reroll_planet = yes
		clear_blockers = yes
	}
	else_if = {
		limit = { is_planet_class = pc_arid }
		change_pc = pc_continental
		reroll_planet = yes
		clear_blockers = yes
	}
	else_if = {
		limit = { is_planet_class = pc_savannah }
		change_pc = pc_tropical
		reroll_planet = yes
		clear_blockers = yes
	}
	else_if = {
		limit = { is_planet_class = pc_ocean }
		change_pc = pc_arctic
		reroll_planet = yes
		clear_blockers = yes
	}
	else_if = {
		limit = { is_planet_class = pc_continental }
		change_pc = pc_alpine
		reroll_planet = yes
		clear_blockers = yes
	}
	else_if = {
		limit = { is_planet_class = pc_tropical }
		change_pc = pc_tundra
		reroll_planet = yes
		clear_blockers = yes
	}
	else_if = {
		limit = { is_planet_class = pc_arctic }
		change_pc = pc_frozen
		reroll_planet = yes
		clear_blockers = yes
	}
	else_if = {
		limit = { is_planet_class = pc_alpine }
		change_pc = pc_frozen
		reroll_planet = yes
		clear_blockers = yes
	}
	else_if = {
		limit = { is_planet_class = pc_tundra }
		random_list = {
			50 = { change_pc = pc_frozen }
			50 = { change_pc = pc_barren_cold }
		}
		reroll_planet = yes
		clear_blockers = yes
	}
}
