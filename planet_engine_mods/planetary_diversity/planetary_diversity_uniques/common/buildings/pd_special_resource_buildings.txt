building_food_processing_center = {
	base_buildtime = 480
	can_build = no
	category = resource
		
	planet_modifier = {
		planet_farmers_food_produces_mult = 0.25
	}
	
	allow = {
		exists = owner
		has_upgraded_capital = yes
	}
	
	destroy_trigger = {
		owner = { is_ai = yes }
		OR = {
			is_planet_class = pc_machine
			has_modifier = resort_colony
		}
		OR = {
			num_districts = { type = district_farming value < 6 }
			num_districts = { type = district_farming_uncapped value < 6 }
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_farmer_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_agri_drone_add = 2
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
			volatile_motes = 50
		}
		upkeep = {
			energy = 1
			volatile_motes = 1
		}
	}	
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_agri_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_farmer_effect_desc
	}
	
	upgrades = {
		building_crystal_pillar
	}
	
	prerequisites = {
		"tech_food_processing_2"
	}	
	
	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			buildings_has_motes = no
		}
		modifier = {
			factor = 0
			owner = { has_country_flag = buildings_new_mote_building }
		}
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
	
	on_built = {
		planet = { 
			owner = { set_timed_country_flag = { flag = buildings_new_mote_building days = 480 } }
		}
	}
}
