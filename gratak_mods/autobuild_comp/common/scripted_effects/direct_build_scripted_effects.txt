direct_build_affects_autobuild_init = {
    owner = { set_country_flag = display_low_tier_flag }
}

direct_build_affects_autobuild_finish = {
    owner = {
		if = {
			limit = { NOT = { has_country_flag = do_no_remove_low_tier_flag } }
            remove_country_flag = display_low_tier_flag
		}
	}
}
