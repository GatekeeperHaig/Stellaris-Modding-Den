check_neighboring_adj_bonus_buildings_ag = {
	if = {
		limit = {
			NAND = {
				exists = owner
				owner = { has_authority = auth_machine_intelligence }
			}
		}
		every_neighboring_tile = {
			if = {
				limit = { has_building = virtual_reality_center }
				prevprev = {
					change_variable = { which = energy_weight value = 1 }
				}
				else = {
					if = {
						limit = { has_building = disease_control_center }
						prevprev = {
							change_variable = { which = society_research_weight value = 2 }
						}
					}
				}
			}
		}
	}
	if = {
		limit = {
			NAND = {
				exists = owner
				owner = {
					NOT = { has_authority = auth_machine_intelligence }
				}
			}
		}
		every_neighboring_tile = {
			if = {
				limit = { has_building = data_stream_amplifier }
				prevprev = {
					change_variable = { which = energy_weight value = 1 }
				}
				else = {
					if = {
						limit = { has_building = biomaterial_bank }
						prevprev = {
							change_variable = { which = society_research_weight value = 2 }
						}
					}
				}
			}
		}
	}
	every_neighboring_tile = {
		if = {
			limit = { has_building = automated_fabrication_complex }
			prevprev = {
				change_variable = { which = minerals_weight value = 1 }
				change_variable = { which = energy_weight value = 1 }
			}
		}
	}
}
check_planet_bonus_buildings_ag = {
	if = {
		limit = {
			NAND = {
				exists = owner
				owner = { has_authority = auth_machine_intelligence }
			}
		}
		#NON_UNIQUE
		every_tile = {
			if = {
				limit = { has_building = museum_luxury }
				prev = {
					change_variable = { which = energy_mult_planet_building value = 0.15 }
					change_variable = { which = society_research_mult_planet_building value = 1 }
				}
				else = {
					if = {
						limit = { has_building = disease_control_center }
						prev = {
							change_variable = { which = society_research_mult_planet_building value = 0.25 }
						}
					}
				}
			}
		}
	}
	if = {
		limit = {
			NAND = {
				exists = owner
				owner = {
					NOT = { has_authority = auth_machine_intelligence }
				}
			}
		}
		#NON_UNIQUE
		every_tile = {
			if = {
				limit = { has_building = garbage_recycling_center }
				prev = {
					change_variable = { which = minerals_mult_planet_building value = 0.15 }
					change_variable = { which = society_research_mult_planet_building value = 0.15 }
					change_variable = { which = engineering_research_mult_planet_building value = 0.10 }
				}
				else = {
					if = {
						limit = { has_building = biomaterial_bank }
						prev = {
							change_variable = { which = society_research_mult_planet_building value = 0.30 }
						}
					}
				}
			}
		}
	}
	#NON_UNIQUE
	every_tile = {
		if = {
			limit = { has_building = automated_fabrication_complex }
			prev = {
				change_variable = { which = minerals_mult_planet_building value = 0.25 }
				change_variable = { which = engineering_research_mult_planet_building value = 0.25 }
			}
		}
	}
}
