### Migrating Forests
planet_event = {
	id = colony.1
	title = "colony.1.name"
	desc = "colony.1.desc"
	picture = GFX_evt_alien_nature
	show_sound = event_alien_nature
	location = ROOT
	trackable = yes
	
	trigger = {
		has_owner = yes
		original_owner = yes
		is_homeworld = no
		owner = { 
			is_ai = no
			NOT = { has_authority = auth_machine_intelligence }
		}
		has_ground_combat = no
		is_occupied_flag = no
		num_pops > 0
		is_capital = no
		OR = {
			is_hot_wet_category_planet = yes 
			is_hot_dry_category_planet = yes
		}
		NOR = { 
			has_global_flag = migrating_forests_global
			has_planet_flag = colony_event
		}
	}

	is_triggered_only = yes
	#mean_time_to_happen = {
	#	months = 640
	#}
	
	immediate = {
		set_global_flag = migrating_forests_global
		set_planet_flag = colony_event
		random_tile = {
			limit = { 
				has_blocker = no
				has_grown_pop = no
				has_building = no
				has_building_construction = no
			}
			set_blocker = "tb_wandering_forests"
		}
		random_tile = {
			limit = { 
				has_blocker = no
				has_grown_pop = no
				has_building = no
				has_building_construction = no
			}
			set_blocker = "tb_wandering_forests"
		}
		random_tile = {
			limit = { 
				has_blocker = no
				has_grown_pop = no
				has_building = no
				has_building_construction = no
			}
			set_blocker = "tb_wandering_forests"
		}
		add_modifier = {
			modifier = "migrating_forests"
			days = -1
		}
	}
	
	option = {
		name = colony.1.a
		begin_event_chain = {
			event_chain = "migrating_forests_chain"
			target = ROOT.owner
		}
		enable_special_project = {
			name = "MIGRATING_FORESTS_1_PROJECT"
			location = this
			owner = root.owner
		}
		enable_special_project = {
			name = "MIGRATING_FORESTS_2_PROJECT"
			location = this
			owner = root.owner
		}
	}
}

### Abandoned Terraforming Equipment
planet_event = {
	id = colony.100
	title = "colony.100.name"
	desc = "colony.100.desc"
	picture = GFX_evt_underground_civilization
	show_sound = event_mystic_reveal
	location = ROOT
	trackable = yes
	
	trigger = {
		has_owner = yes
		original_owner = yes
		is_homeworld = no
		owner = { is_ai = no }
		has_ground_combat = no
		is_occupied_flag = no
		num_pops > 0
		is_capital = no
		NOR = {
			has_planet_flag = abandoned_terraforming_planet
			has_planet_flag = colony_event
			is_planet_class = pc_nuked
			is_planet_class = pc_gaia
			is_planet_class = pc_ringworld_habitable
			is_planet_class = pc_habitat
			owner = { has_country_flag = abandoned_terraforming_country }
			owner = { has_authority = auth_machine_intelligence }
			is_ideal_planet_class = { who = owner status = yes }
		}
	}
	
	is_triggered_only = yes
	#mean_time_to_happen = {
	#	months = 640
	#}
	
	immediate = {
		set_planet_flag = abandoned_terraforming_planet
		set_planet_flag = colony_event
		owner = { set_country_flag = abandoned_terraforming_country }
	}
	
	option = {
		name = colony.100.a
		hidden_effect = {
			add_modifier = {
				modifier = "abandoned_terraforming"
				days = -1
			}
		}
		begin_event_chain = {
			event_chain = "abandoned_terraforming_chain"
			target = ROOT.owner
		}
		enable_special_project = {
			name = "ABANDONED_TERRAFORMING_1_PROJECT"
			location = this
			owner = root.owner
		}
		enable_special_project = {
			name = "ABANDONED_TERRAFORMING_2_PROJECT"
			location = this
			owner = root.owner
		}
	}
}
