event = {
	id = fleetcap.1000
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_country = {
			limit = {
				is_country_type = default
				is_subject = no
				NOT = {
					has_country_flag = has_fleetcap_modifier
				}
			}
			add_modifier = {
				modifier = "fleetcap_modifier1"
				days = -1
			}
			set_country_flag = has_fleetcap_modifier
		}
	}
}
