	on_unqueued = { subtract_food_buildings = yes }

		has_resource = no

		weight = 10

			factor = 5
			from.owner = { has_country_flag = food_focus_factor_1 }
		}
		modifier = { 
			factor = 10
			from.owner = { has_country_flag = food_focus_factor_2 }
		}	
	}
}
