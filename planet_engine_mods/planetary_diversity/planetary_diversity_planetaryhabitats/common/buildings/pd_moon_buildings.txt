

building_loadingbays = {
	base_buildtime = 360
	base_cap_amount = 1

	category = resource
	
	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
		OR = {
		is_planet_class = pc_tmhab
		is_planet_class = pc_tmhab2
		}
	}

	allow = {
		buildings_simple_allow = yes
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_merchant_add = 2
			job_clerk_add = 6
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
		}
		upkeep = {
			energy = 5
		}
	}	
	
	ai_weight = {
		weight = 10	
	}
}
