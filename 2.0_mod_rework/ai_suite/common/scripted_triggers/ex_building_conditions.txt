### Building Conditions

conditions_power_planet_1 = {
	planet_is_habitat = no
	OR = {
		owner_has_tech_power_planet_2 = no
		planet = { 
			OR = {
				has_capital_1 = no
				has_capital_2 = no
				has_capital_3 = no
			}	
		}
	}		
}

####### Destroy Conditions

standard_unity_building_destroy_conditions = {
	if = {
		limit = { overhauled_building_content_enabled = no }
		AND = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
	}	
	if = {
		limit = { overhauled_building_content_enabled = yes }
		AND = {
			exists = owner
			owner = { is_gestalt_consciousness = yes }
		}
	}	
}
