	if = {
		limit = {
			OR = { has_building_construction = yes has_building = yes }
		}
		owner = { set_country_flag = cgm_auto_built }
		else = {
			if = {
				limit = {
					owner = { has_swapped_tradition = tr_purity_symbol_purity }
					NOR = { has_building = building_symbol_purity has_building = building_eutab_monument_purity }
					owner = {
						check_variable = { which = cgm_special_bestWeight value < 10 }
					}
				}
				prev = {
					set_variable = { which = cgm_special_bestWeight value = 10 }
					set_variable = { which = cgm_special_bestBuilding value = -2111778.139 }
				}
				save_event_target_as = cgm_best_planet_for_special
			}
			if = {
				limit = {
					OR = { has_building_construction = yes has_building = yes }
				}
				owner = { set_country_flag = cgm_auto_built }
				else = {
					if = {
						limit = {
							owner = { has_swapped_tradition = tr_synchronicity_hive_mind_synapse }
							NOR = {
								has_building = building_hive_synapse
								has_building = building_eutab_hive_neuron_cluster
								has_building = building_eutab_hive_alpha_node
							}
							owner = {
								check_variable = { which = cgm_special_bestWeight value < 10 }
							}
						}
						prev = {
							set_variable = { which = cgm_special_bestWeight value = 10 }
							set_variable = { which = cgm_special_bestBuilding value = -2078223.707 }
						}
						save_event_target_as = cgm_best_planet_for_special
					}
					if = {
						limit = {
							OR = { has_building_construction = yes has_building = yes }
						}
						owner = { set_country_flag = cgm_auto_built }
						else = {
							if = {
								limit = {
									owner = {
										AND = {
											has_non_swapped_tradition = tr_diplomacy_alien_tourism
											NOR = { has_ethic = ethic_xenophobe has_ethic = ethic_fanatic_xenophobe }
										}  #eutab
									}
									NOR = { has_building = building_visitor_center has_building = building_eutab_xeno_district }
									owner = {
