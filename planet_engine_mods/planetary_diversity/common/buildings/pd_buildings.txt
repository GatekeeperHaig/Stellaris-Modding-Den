building_eludium_plant = {
	base_buildtime = 360

	category = resource
	
	potential = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
		OR = {
			NOT = { is_planet_class = pc_habitat }
			AND = {
				exists = owner
				owner = { is_gestalt = yes }
			}
		}
	}

	base_cap_amount = 1
	
	planet_modifier = {
		planet_technician_energy_produces_mult = 0.10
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_technician_add = 4
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_technician_drone_add = 4
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			district_generator_max = 2
			planet_technician_energy_produces_mult = 0.05
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		cost = {
			trigger = {
				owner = { 
					is_machine_empire = yes
				}
			}
			minerals = 100
			alloys = 100
		}
		upkeep = {
			sr_eludium = 1
		}
	}	
	
	prerequisites = {
		tech_eludium_plant
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_technician_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_technician_effect_desc
	}
	
	ai_weight = {
		weight = 2
		modifier = 
		{
			factor = 5
			owner = { 
				has_monthly_income = {
					resource = sr_eludium
					value < 2
				}
			}
		}
		# No one to work it
		modifier = {
			factor = 0
			count_owned_pops = { 
				limit = { 
					OR = { 
						is_unemployed = yes 
						has_job = servant 
					}
				}
				count < 1
			}
		}
	}

	ai_resource_production = {
		energy = 1
		trigger = {
			always = yes
		}
	}
}
