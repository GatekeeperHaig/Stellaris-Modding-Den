namespace = ex_new_worlds_spawn_engine

event = { 
	id = ex_new_worlds_spawn_engine.1
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes
	
	#This event fires only once.
		
	immediate = {		
		every_planet = { 
			if = {
				limit = {
					is_colonizeable = yes
					has_pop = yes					
				}	
				planet_event = { id = ex_new_worlds_spawn_engine.3 days = 1 }
			}
			if = {
				limit = {
					is_colonizeable = yes
					is_colony = no
					has_owner = no
					has_pop = no					
				}	
				planet_event = { id = ex_new_worlds_spawn_engine.3 days = 2 }					
			}							
			if = {
				limit = {
					is_colonizeable = no
					is_star = no
					is_asteroid = no
				}
				planet_event = { id = ex_new_worlds_spawn_engine.4 days = 3 }					
			}	
		}
	}
}

planet_event = { 
	id = ex_new_worlds_spawn_engine.2
	hide_window = yes
	is_triggered_only = yes
	
	
	immediate = { 
		ex_new_worlds_habitable_planets_spawn_table = yes
		ex_new_worlds_replace_planet_pref_traits = yes
	}
}

	is_triggered_only = yes
	
	
	immediate = { ex_new_worlds_habitable_planets_spawn_table = yes }
}
