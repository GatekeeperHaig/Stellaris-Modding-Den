#  You can of course many triggers in the modifier = { factor = int <trigger> } field for the random list. Below is a list of custom triggers
#
#    has_moon = yes/no
#    is_moon = yes/no
#    is_gas_giant_moon = yes/no
#    is_inside_nebula = yes/no
#    has_ring = yes/no
#	 regular_star_system = yes/no - means the system has a single star
#	 binary_star_system = yes/no - 2 stars in the system
#	 ternary_star_system = yes/no - 3 stars in the system
