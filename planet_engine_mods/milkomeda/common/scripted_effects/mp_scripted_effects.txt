# Destroys a random hyperlane from this system to a neighbouring one
# THIS = system
mp_destroy_hyperlane = {
	random_neighbor_system = {
		limit = {
			NOR = {
				has_star_flag = "empire_home_system"
				has_star_flag = "marauder_system"
				any_system_planet = { has_planet_flag = "fallen_empire_world" }
			}
		}
		#ignore_hyperlanes = no
		remove_hyperlane = { 
			from = PREV
			to = THIS 
		}
	}
}

mp_create_hyperlane = {
	if = {
		limit = {
			any_neighbor_system = { 
				ignore_hyperlanes = yes
				NOT = {
					has_hyperlane_to = PREV
				}
			}
		}
		
		random_neighbor_system = {
			ignore_hyperlanes = yes
			limit = {
				NOR = {
					has_hyperlane_to = PREV
					has_star_flag = "spawned_hyperlanes"
				}
			}
			
			add_hyperlane = { 
				from = PREV
				to = THIS 
			}
		}
	}
}
