building_nutrient_paste_facility = {
	base_buildtime = @resource_building_build_time_tier_2
	cost = { minerals = @resource_building_mineral_cost_tier_2 }
	required_resources = { energy = @resource_building_upkeep_tier_2 }
	produced_resources = { food = 5 }
	
	show_tech_unlock_if = {
		has_authority = auth_machine_intelligence
		has_government = gov_machine_servitor # required civic cannot be added/removed
	}
	
	potential = {
		owner = {
			has_authority = auth_machine_intelligence
			has_government = gov_machine_servitor
		}
	}
	
	destroy_if = {
		exists = owner
		owner = {
			NOT = {
				has_authority = auth_machine_intelligence
				has_government = gov_machine_servitor
			}
		}
	}
	
	ai_allow = {
		tile = {
			OR = {
				has_deposit = no
				AND = {
					has_food = yes
					has_vanilla_or_exoverhaul_strategic_resources = no
				}
			}
		}
	}
	ai_weight = { weight = @important }
}


building_basic_power_plant = {
	base_buildtime = @resource_building_build_time_tier_0
	
	cost = {
		minerals = @resource_building_mineral_cost_tier_0
	}
	
	potential = {
		planet = { has_capital_0 = yes }
	}
	produced_resources = { energy = @resource_building_output_tier_0 }
	
	upgrades = {
		building_power_plant_1
	}
	
	ai_allow = {
		tile = {
			OR = {
				AND = {
					has_energy = yes
					has_vanilla_or_exoverhaul_strategic_resources = no
				}
				AND = {
					has_vanilla_or_exoverhaul_strategic_resources = no
					has_energy = no
					has_minerals = no
					has_research_resources = no
				}
			}
		}
	}
	
	ai_weight = {
		weight = @standard
		modifier = {
			factor = @good
			tile = {
				has_resource = {
					type = energy
					amount = 1
				}
			}
		}
		modifier = {
			factor = @important
			tile = {
				has_resource = {
					type = energy
					amount = 2
				}
			}
		}
		modifier = {
			factor = @strategic
			tile = {
				has_resource = {
					type = energy
					amount > 2
				}
			}
		}
		modifier = {
			factor = @reduce_interest_1
			tile = {
				OR = {
					has_energy = no
					AND = {
						owner_is_not_machine_or_is_servitor = yes
						has_high_food = yes
					}
				}
			}
		}
		modifier = {
			factor = @reduce_interest_2
			tile = {
				has_slave_pop = yes
				has_prole_or_stapled_pop = yes
				AND = {
					has_non_sapient_robot = yes
					exists = owner
					owner = { NOT = { has_global_flag = "ex_robots_active" } }
				}
			}
		}
		modifier = {
			factor = @increase_interest_1
			tile = { has_energy_bonus_pop = yes	}
		}
		modifier = {
			factor = @increase_interest_1
			energy_sr_adj_bonus_building = yes			
		}
		modifier = {
			factor = @increase_interest_1
			energy_sr_bonus_building = yes			
		}		
		modifier = {
			factor = @increase_interest_1
			tile = { any_neighboring_tile = { has_hyperstorage_building = yes } }
		}
		modifier = {
			factor = @increase_interest_1
			tile = { any_neighboring_tile = { has_capital_building = yes } }
		}
		modifier = {
			factor = @increase_interest_1
			tile = {
				OR = {
					AND = {
						has_robot_pop = yes
						exists = owner
						owner = { has_global_flag = "ex_robots_active" }
					}
					has_free_sapient_robot = yes
				}
			}
		}
		modifier = {
			factor = @increase_interest_1
			tile = { owner = { has_authority = auth_machine_intelligence } }
		}
		modifier = {
			factor = @increase_interest_1
			planet = { has_modifier = "strong_magnetic_field" }
		}
		modifier = {
			factor = @increase_interest_3
			planet = { has_modifier = "hazardous_weather" }
		}
		modifier = {
			factor = @crucial_1
			owner = { has_country_flag = is_energy_starved }
		}
	}
}
