# Possible Tile Blocker Variables
# spawn_chance, tile scope, value will compete with the spawn chance of every other item and there's a constant 100 for no blocker to spawn.
# cost, resource table, the cost to clear this tile blocker
# time, amount of days it takes to clear this tile blocker.
# picture = tb_active_volcano #Use art from tb_active_volcano

# Variables
@buckets = 128
@loads = 64
@lot = 32
@enough = 16
@some = 8
@little = 4
@tiny = 2


tb_persistent_volcano = {
  picture = tb_active_volcano
  cost = { # Can't be removed
  }

	spawn_chance = {
    modifier = {
			add = @lot
			planet = {
				is_colonizable = yes
				has_modifier = "young_world"
			}
		}
		modifier = {
	    add = @some
	    planet = {
	      is_colonizable = yes
				OR = {
					is_planet_class = pc_geothermal
          has_modifier = "unstable_tectonics"
				}
			}
	  }
    modifier = {
      add = @little
      planet = {
        is_colonizable = yes
				is_planet_class = pc_desertislands
      }
    }
		modifier = {
	    add = @tiny
	    planet = {
	      is_colonizable = yes
				NOT = { is_planet_class = pc_floating }
				NOT = { is_planet_class = pc_crystal }
				NOT = { is_planet_class = pc_city }
				NOT = { is_planet_class = pc_geothermal }
			}
	  }
	  modifier = {
	    factor = 0.25
	    planet = {
	      has_modifier = "titanic_life"
	    }
	  }
	  modifier = {
	      factor = 0.2
	      planet = {
	        has_modifier = "boundless_plains"
	      }
	  }
	  modifier = {
	    factor = 0.25
	    planet = {
	      planet_size < 15
	    }
	  }
	}

}
