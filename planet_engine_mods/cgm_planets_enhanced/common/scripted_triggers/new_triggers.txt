allows_robots = { 
	OR = { 
		AND = { 
			is_ai = no 
			OR = { 
				has_policy_flag = robots_allowed
				is_machine_empire = yes				
				is_synth_empire = yes
			}	
		}
		AND = { 
			is_ai = yes
			OR = { 
				AND = { 
					has_policy_flag = robots_allowed
					has_ai_personality_behaviour = robot_exploiter
				}
				AND = { 
					has_policy_flag = robots_allowed
					has_ai_personality_behaviour = robot_liberator	
				}	
				is_machine_empire = yes				
				is_synth_empire = yes	
			}
		}
	}	
}

owner_allows_robots = { 
	exists = owner
	owner = { allows_robots = yes }
}

planet_has_robots = { 
	planet = { any_owned_pop = { is_robot_pop = yes } }
}
