autobuild_trigger_fleet_is_valid = {
	exists = owner
	is_ship_class = shipclass_constructor
	is_disabled = no
	is_in_combat = no
	is_fleet_idle = yes
	#sector_controlled = no
	# try to filter out non sector constructors
	exists = solar_system
	solar_system = {
		not = {
			any_planet = { sector_controlled = yes }
		}
	}
	not = {
		has_fleet_flag = autobuild_fleet
		has_fleet_flag = autonomous_assembler
	}
}
autobuild_trigger_fleet_valid_planet = {
	has_owner = no
	has_orbital_station = no
	has_any_megastructure = no
	nor = {
		has_planet_flag = autobuild_planet_block
		has_planet_flag = ass_target
	}
	is_surveyed = {
		who = root.owner
		status = yes
	}
	or = {
		has_deposit_for = shipclass_research_station
		and = {
			has_deposit_for = shipclass_mining_station
			or = {
				orbital_deposit_tile = {
					has_resource = {
						type = energy
						amount > 0
					}
				}
				nor = {
					and = {
						root.owner = { has_country_flag = autobuild_ship_ignore_2_and_less_mineral }
						orbital_deposit_tile = {
							has_resource = {
								type = minerals
								amount < 3
							}
						}
					}
					and = {
						root.owner = { has_country_flag = autobuild_ship_ignore_1_and_less_mineral }
						orbital_deposit_tile = {
							has_resource = {
								type = minerals
								amount < 2
							}
						}
					}
				}
			}
		}
	}
}

autobuild_trigger_tile_has_pop = {
	exists = pop
	pop = {
		exists = owner
		or = {
			is_being_purged = no
			has_purge_type = {
				#country = owner
				type = purge_neutering
			}
		}
		nand = {
			has_citizenship_type = {
				#country = owner
				type = citizenship_slavery
			}
			or = {
				has_slavery_type = {
					#country = owner
					type = slavery_livestock
				}
				has_slavery_type = {
					#country = owner
					type = slavery_matrix
				}
			}
		}
	}
}

			}
		}
	}

	has_pop = no
	root.owner = { has_country_flag = autobuild_local_resettle }
	root = {
		any_tile = {
			or = {
				has_building = no
				autobuild_trigger_has_any_regular_building = yes
			}
			autobuild_trigger_tile_has_pop = yes
		}
	}
}

	}
