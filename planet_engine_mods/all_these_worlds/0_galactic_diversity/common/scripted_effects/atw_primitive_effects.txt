create_temperate_species = {
	if = {
		limit = { is_planet_class = pc_ocean }
		# Convert placeholder to a Pelagic class
		# but start with the modifiers first, knowing we want a pelagic

		# Spawn Lithosphere modifier
		add_modifier = {
			modifier = "lithosphere-oceanic"
			days = -1
		}
		# Spawn Hydrosphere modifier
		add_modifier = {
			modifier = "hydrosphere-water_pelagic"
			days = -1
		}
		# Spawn Biosphere modifier
		add_modifier = {
			modifier = "biosphere-aquatic"
			days = -1
		}
		# Spawn Atmosphere modifier
		add_modifier = {
			modifier = "atmosphere-optimal"
			days = -1
		}
		# Spawn Gravity modifier
		spawn_gravity_modifier = yes
		# Create species before changing planet class
		create_amphibian_species = yes
		# Change pc
		change_pc = "pc_archipelagic"
		set_planet_entity = { picture = biome_archipelagic_lush entity = archipelagic_planet_01_entity }
		set_planet_flag = "planet_converted"		
	}
	if = {
		limit = { is_planet_class = pc_continental }
		# Convert placeholder to an abundant water class
		# in this case, start by picking a random lithosphere that will decide which pc we get

		# Spawn Lithosphere modifier
		random_list = {
			1 = { # This one will result in Pangaian
				add_modifier = {
					modifier = "lithosphere-oceanic"
					days = -1
				}
				# Spawn Hydrosphere modifier
				add_modifier = {
					modifier = "hydrosphere-water_abundant"
					days = -1
				}
				# Spawn Biosphere modifier
				add_modifier = {
					modifier = "biosphere-dense"
					days = -1
				}
				# Spawn Atmosphere modifier
				add_modifier = {
					modifier = "atmosphere-optimal"
					days = -1
				}
				# Spawn Gravity modifier
				spawn_gravity_modifier = yes
				# Create species before changing planet class
				create_arboreal_species = yes
				# Change pc
				change_pc = "pc_pangaian"
				set_planet_entity = { picture = biome_pangaian_lush entity = gd_pangaian_planet_lush_01_entity }
				set_planet_flag = "planet_converted"
			}
			2 = { # This one will result in Chlorogenian
				add_modifier = {
					modifier = "lithosphere-continental"
					days = -1
				}
				# Spawn Hydrosphere modifier
				add_modifier = {
					modifier = "hydrosphere-water_abundant"
					days = -1
				}
				# Spawn Biosphere modifier
				add_modifier = {
					modifier = "biosphere-diverse"
					days = -1
				}
				# Spawn Atmosphere modifier
				add_modifier = {

					days = -1
				}
				# Spawn Gravity modifier

				# Create species before changing planet class

				# Change pc

				set_planet_flag = "planet_converted"

			1 = { # This one will result in Wetland
