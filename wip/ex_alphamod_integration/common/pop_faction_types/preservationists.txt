preservationists = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_xenophile
	
	unique = yes
	support_multiplier = 1
	
	is_potential = {		
		NOT = { has_ethic = ethic_gestalt_consciousness }
		NOT = { has_modifier = banned_xenophile }
		NOR = {
			has_ethic = ethic_fanatic_authoritarian
			has_ethic = ethic_fanatic_xenophobe
			has_ethic = ethic_fanatic_militarist
			has_ethic = ethic_militarist
		}
		has_technology = tech_alpha_traditional_farming
		OR = {
			has_technology = tech_alpha_preservations
			has_technology = tech_alpha_food_hybrids
			has_technology = tech_atmospheric_manipulation
			has_country_flag = cf_is_terraformer
		}
	}
	
	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}
	
	can_join_faction = {
		is_enslaved = no
		is_being_purged = no
		has_culture_shock = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_ethic = ethic_pacifist
			has_ethic = ethic_fanatic_pacifist
			has_ethic = ethic_spiritualist
			has_ethic = ethic_fanatic_spiritualist
			has_ethic = ethic_xenophile
			has_ethic = ethic_fanatic_xenophile
		}
		is_robot_pop = no
		tile = {
			nor = {
				has_resource = minerals
				has_resource = engineering_research				
			}
			OR = {
				has_resource = food
				has_resource = society_research
				has_resource = sr_biomass
				has_resource = sr_natural_fuels
			}
		}
	}
	
	attraction = {
		base = 180
		
		modifier = {
			factor = 0.25
			parameter:empire = {
				has_modifier = banned_xenophile
			}						
		}
		
		modifier = {
			factor = 1.5
			owner = {
				has_ethic = ethic_xenophile
				has_ethic = ethic_fanatic_xenophile			
			}
		}
		
		modifier = {
			factor = 0
			exists = from
			from = {
				has_modifier = prosecuted_faction
			}
		}

		modifier = {
			factor = 0.5
			owner = {		
				has_policy_flag = alp_private_businesses_prohibited
			}
		}
		modifier = {
			factor = 0.8
			owner = {		
				has_policy_flag = alp_private_businesses_regulated
			}
		}
		
		modifier = {
			factor = 1.5
			OR = {
				has_ethic = ethic_xenophile
				has_ethic = ethic_fanatic_xenophile
				has_ethic = ethic_spiritualist
				has_ethic = ethic_fanatic_spiritualist
			}
		}
		
		modifier = {
			factor = 0.25
			OR = {
				has_ethic = ethic_materialist
				has_ethic = ethic_fanatic_materialist
			}
		}
		
		modifier = {
			factor = 1.25
			tile = {
				has_resource = sr_biomass
			}
		}
		
		modifier = {
			factor = 1.25
			tile = {
				has_resource = sr_exotic_foods
			}
		}
		
		modifier = {
			factor = 1.5
			planet = {
				or = {
					has_planet_flag = pf_has_been_terraformed
					is_planet_class = pc_gaia
				}
			}
		}
		
		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}
	}
	
	leader = {
		base = 100
		modifier = {
			factor = 0
			is_event_leader = yes
		}	
	}
	
	demand = {
		title = "TOTALITARIAN_REGULATED_BUSINESS"
		unfulfilled_title = "TOTALITARIAN_NOT_REGULATED_BUSINESS"
		desc = "TOTALITARIAN_REGULATED_BUSINESS_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -5

		potential = {
			exists = owner
			owner = {
				not = { has_policy_flag = alp_private_businesses_prohibited }
			}
		}

		trigger = {
			owner = {
				has_policy_flag = alp_private_businesses_regulated
			}
		}
	}
	
	demand = {
		title = "PRESERVERS_NO_FRACKING"
		unfulfilled_title = "PRESERVERS_FRACKING"
		desc = "PRESERVERS_NO_FRACKING_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -15

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				not = {
					any_owned_planet = {
						has_planet_flag = pf_fracking
					}
				}
			}
		}
	}
	
	demand = {
		title = "PRESERVERS_NO_TERRAFORMING"
		unfulfilled_title = "PRESERVERS_TERRAFORMING"
		desc = "PRESERVERS_NO_TERRAFORMING_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = {
				has_technology = tech_ecological_adaptation
			}
		}

		trigger = {
			owner = {
				not = {
					has_country_flag = cf_recent_terraforming
				}
			}
		}
	}
	
	demand = {
		title = "PRESERVERS_RENEWABLE_ONLY"
		unfulfilled_title = "PRESERVERS_NOT_RENEWABLE_ONLY"
		desc = "PRESERVERS_RENEWABLE_ONLY_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -5

		potential = {
			exists = owner
			owner = {
				any_owned_planet = {
					any_tile = {
						has_resource = sr_natural_fuels
					}
				}
			}
		}

		trigger = {
			owner = {
				not = {
					any_owned_planet = {
						or = {
							has_building = building_fuel_sourcer
							has_building = building_fuel_sourcer_2
							has_building = building_fuel_sourcer_3
						}
					}
				}
			}
		}
	}
	
	demand = {
		title = "PRESERVERS_NO_BIOMASS_EXPLOITATION"
		unfulfilled_title = "PRESERVERS_BIOMASS_EXPLOITATION"
		desc = "PRESERVERS_NO_BIOMASS_EXPLOITATION_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -15

		potential = {
			exists = owner
			owner = {
				any_owned_planet = {
					any_tile = {
						has_resource = sr_biomass
					}
				}
			}
		}

		trigger = {
			owner = {
				not = {
					any_owned_planet = {
						or = {
							has_building = building_biomass_sourcer
							has_building = building_biomass_sourcer_2
							has_building = building_biomass_sourcer_3
							has_building = building_biomass_furnace
						}
					}
				}
			}
		}
	}
	
	demand = {
		title = "PRESERVERS_PREFER_TRADITIONAL_FARMING"
		unfulfilled_title = "PRESERVERS_NOT_PREFER_TRADITIONAL_FARMING"
		desc = "PRESERVERS_PREFER_TRADITIONAL_FARMING_DESC"

		fulfilled_effect = 5

		potential = {
			exists = owner
			owner = {
				any_owned_planet = {
					or = {
						has_building = building_hydroponics_farm_1
						has_building = building_hydroponics_farm_2
						has_building = building_hydroponics_farm_3
						has_building = building_hydroponics_farm_4
					}
				}
			}
		}

		trigger = {
			owner = {
				not = {
					any_owned_planet = {
						or = {
							has_building = building_food_hybridiser
							has_planet_flag = pf_non_traditional_farming_preference
						}
					}
				}
			}
		}
	}
	
	on_create = {
		pop_faction_event = { id = factions.1 }
	}

	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_xenophile
			}
			remove_modifier = promoted_xenophile
		}
		if = {
			limit = {
				has_modifier = suppressed_xenophile
			}
			remove_modifier = suppressed_xenophile
		}
		if = {
			limit = {
				has_modifier = banned_xenophile
			}
			remove_modifier = banned_xenophile
		}
	}
	
	actions = {	
		prosecute_faction = {
			title = "PROSECUTE_FACTION"
			description = "PROSECUTE_FACTION_DESC"
			
			potential = {
				exists = owner
				not = { has_modifier = prosecuted_faction }
				owner = {
					or = {
						has_modifier = suppressed_xenophile
						has_modifier = banned_xenophile
					}
				}
			}

			effect = {
				add_modifier = { modifier = prosecuted_faction days = -1 }
			}
			
			ai_weight = {
				base = 0
			}			
		}		
		cancel_prosecute_faction = {
			title = "CANCEL_PROSECUTE_FACTION"
			description = "PROSECUTE_FACTION_DESC"
			
			potential = {
				exists = owner
				has_modifier = prosecuted_faction
			}	

			effect = {
				remove_modifier = prosecuted_faction
			}
			
			ai_weight = {
				base = 0
			}			
		}
		
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				owner = {
					NOR = {
						has_modifier = suppressed_xenophile
						has_modifier = promoted_xenophile
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = suppressed_xenophile days = -1 }
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"
			
			potential = {
				exists = owner
				owner = {
					has_modifier = suppressed_xenophile
				}
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_xenophile
				}
			}
			
			ai_weight = {
				base = 0
			}
		}
	}
