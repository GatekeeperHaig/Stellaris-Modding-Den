#EnterSystem
ship_event = {
	id = pdbiolumen.10
	title = pdbiolumen.10.name
	desc = pdbiolumen.10.desc
	picture = GFX_evt_biolumen
	location = from
	show_sound = event_ship_bridge

	is_triggered_only = yes

	trigger = {
		from = {
			has_star_flag = biolumen_system
		}
		owner = {
			NOT = {
				has_country_flag = biolumen_encounter
			}
			OR = {
				is_country_type = default
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
			}
		}
	}

	immediate = {
		owner = {
			set_country_flag = biolumen_encounter
		}
	}

	option = {
		name = pdbiolumen.10.a
	}
}

#EventStart
planet_event = {
	id = pdbiolumen.30
	title = pdbiolumen.30.name
	desc = pdbiolumen.30.desc
	picture = GFX_evt_biolumen2
	show_sound = event_alien_nature
	location = root

	is_triggered_only = yes

	trigger = {
		has_planet_flag = biolumen_planet
		has_owner = yes
		is_homeworld = no		
		owner = {
			NOT = {
				has_authority = auth_machine_intelligence
			}
		}
		has_ground_combat = no
		is_occupied_flag = no
	}

	immediate = {
		owner = {
			set_country_flag = biolumen_planet_event
		}
	}

	option = {
		name = pdbiolumen.30.a
		hidden_effect = {
			planet_event = { id = pdbiolumen.31 days = 180 random = 1 }
		}
	}

	option = {
		name = pdbiolumen.30.b
