namespace = planets_enhanced_unique_resources

planet_event = { #aurantiacum
	id = planets_enhanced_unique_resources.1
	is_triggered_only = yes
	title = "planets_enhanced_unique_resources.1.name"
	desc = "planets_enhanced_unique_resources.1.desc"
	picture = GFX_evt_underground_civilization
	show_sound = event_mystic_reveal
	location = root	
	
	trigger = {		
		exists = owner		
		owner = { NOT = { has_country_flag = has_aurantiacum } }
		has_resource = { type = sr_aurantiacum_x amount > 0 }
	}	

	option = {
		name = planets_enhanced_unique_resources.1.option
		enable_special_project = {
			name = "AURANTIACUM_PROJECT"
			location = this
			owner = root
		}		
	}
}
