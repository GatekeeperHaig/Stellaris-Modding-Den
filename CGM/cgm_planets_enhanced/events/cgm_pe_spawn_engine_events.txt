country_event = { 
	id = cgm_pe_spawn_engine.0
	hide_window = yes
	
	trigger = { 
		NOR = { 
			has_global_flag = planets_enhanced_active 
			has_global_flag = planets_enhanced_beta_active 
		}
		is_ai = no
	}
	immediate = { 
		set_global_flag = planets_enhanced_active
		if = { 
			limit = { NOT = { has_country_flag = cgm_game_host } }
			set_country_flag = cgm_game_host	
		}	
		every_megastructure = {
			limit = { 
				OR = {
					is_megastructure_type = dyson_sphere_ruined
					is_megastructure_type = think_tank_ruined
					is_megastructure_type = spy_orb_ruined
					is_megastructure_type = gateway_ruined
				}
				exists = orbit				
			}
			orbit = { set_planet_flag = has_ruined_megastructure } 
		}
		every_planet = {
			limit = { 
				is_gas_giant = yes
				has_moon = yes
				any_moon = { NOT = { has_planet_flag = gas_giant_moon } }
			}				
			every_moon = { 
				limit = { NOT = { has_planet_flag = gas_giant_moon } }
				set_planet_flag = gas_giant_moon
			}
		}	
		every_planet = {
			limit = { is_star = yes }
			#this effect block identifies star order in systems				
			if = {
				limit = {
					solar_system = {
						NOT = {
							any_planet = {
								NOT = { is_same_value = prevprev }
								has_planet_flag = system_star_1
							}
						}
					}
				}
				set_planet_flag = system_star_1
			}	
			else_if = {
				limit = {
					solar_system = {
						NOT = {
							any_planet = {
								NOT = { is_same_value = prevprev }
								has_planet_flag = system_star_2
							}
						}
					}
				}
				set_planet_flag = system_star_2
			}	
			else = { set_planet_flag = system_star_3 }
		}				
		random_planet = { 
			limit = { is_star = yes }
			save_global_event_target_as = cgm_pe_variable_storage
			set_variable = { which = tb_sunflower_spawn value = 0 }			
			set_variable = { which = tb_precursor_trading_outpost value = 0 } 
			set_variable = { which = num_of_regular_habitable_planets value = 0 }			
			every_planet = { 
				limit = { habitable_planet = yes } 
				prev = { change_variable = { which = num_of_regular_habitable_planets value = 1 } }
			}		
			set_variable = { which = num_of_asteroidal_candidates value = num_of_regular_habitable_planets }
			divide_variable = { which = num_of_asteroidal_candidates value = 100 }			
			multiply_variable = { which = num_of_asteroidal_candidates value = 15 }			
		}	
		set_default_configuration = yes 
		set_country_flag = level_0_menu	
		country_event = { id = cgm_pe_spawn_engine.1 }	
		country_event = { id = cgm_pe_test_events.1 days = 3 }	
	}
}
