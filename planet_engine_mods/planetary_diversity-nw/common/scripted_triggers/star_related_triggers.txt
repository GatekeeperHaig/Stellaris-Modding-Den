exists_collapsed_star = { # This one is a bit different, the trigger should check if there is any type of collapsed star in the system, not just the primary star, can be used for spawning broken worlds or special resources in systems with a collapsed star present
	OR = {
		solar_system = { any_system_planet = { is_planet_class_pc_black_hole = yes } }
		solar_system = { any_system_planet = { is_planet_class_pc_neutron_star = yes } }
		solar_system = { any_system_planet = { is_planet_class_pc_pulsar = yes } }
		#Realspace
		solar_system = { any_system_planet = { is_planet_class_pc_collapsar = yes } }
		solar_system = { any_system_planet = { is_planet_class_pc_magnetar = yes } }
	}
}

is_collapsed_star = { # This checks if the object is a collapsed star, used for spawning special deposits for instances
	OR = {
		is_planet_class_pc_black_hole = yes
		is_planet_class_pc_neutron_star = yes
		is_planet_class_pc_pulsar = yes
	}
}
