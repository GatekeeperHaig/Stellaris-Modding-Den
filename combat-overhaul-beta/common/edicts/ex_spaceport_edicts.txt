planet_edict = {
	name = "spaceport_automation_economic_specialized"
	influence_cost = 0
	
	effect = {
		planet = {
			hidden_effect = {
				remove_planet_flag = spaceport_automation_economic_specialized				
				remove_planet_flag = spaceport_automation_economic_mixed
				remove_planet_flag = spaceport_automation_stardock_specialized
				remove_planet_flag = spaceport_automation_defensive_specialized
				remove_planet_flag = spaceport_automation_defensive_mixed			
				set_planet_flag = spaceport_automation_economic_specialized							
				set_timed_planet_flag = { flag = reset_delay days = 360 }
				#planet_event = { id = ex_spaceport_events.100 }				
			}
		}
	}	
	
	potential = { 		
		owner = { has_policy_flag = core_edicts }			
		planet = {
			AND = {								
				OR = {
					has_spaceport = yes
					has_spaceport_construction = no
				}
				NOT = { has_planet_flag = reset_delay }	
			}
		}	
	}
	
	ai_weight = {
		weight = 5
		modifier = {
			factor = 0.25
			is_capital = yes
		}			
		#modifier = {
		#	factor = 10 
		#	is_slave_robot_or_prole_planet = yes			
		#}		
	}	
}
