event = {
	id = extended_land_survey.35
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_country = {
			limit = {
				has_country_edict = followup_surveys
				not = { has_country_flag = cf_no_followupsurveyspam2 }
				check_variable = {
					which = "cv_activefollowups" 
					value < 4
				}
			}
			this = { save_event_target_as = country_of_interest }
			random_system = {
				limit = {
					is_inside_border = event_target:country_of_interest
				}
				random_system_planet = {
					limit = {
						is_inside_border = event_target:country_of_interest
						is_surveyed = {
							who = event_target:country_of_interest
							status = yes
						}
						OR = {
							is_planet_class = pc_barren
							is_planet_class = pc_barren_cold
							is_planet_class = pc_asteroid
						}
						not = { has_modifier = terraforming_candidate }
						not = { has_planet_flag = pf_marked_uncolonisable }
						not = { has_planet_flag = pf_already_selected }
					}
						
					event_target:country_of_interest = {
						set_timed_country_flag = {
							flag = cf_no_followupsurveyspam2
							days = 185
						}
					}
						
					random_list = {
						10 = {
							set_planet_flag = pf_marked_uncolonisable
						}
						15 = {
							event_target:country_of_interest = {
								set_timed_country_flag = {
									flag = cf_no_followupsurveyspam2
									days = 725
								}
								change_variable = {
									which = "cv_activefollowups" 
									value = 1
								}	
								country_event = {
									id = extended_land_survey.6
								}
							}
							set_planet_flag = pf_already_selected
							enable_special_project = {
								name = "FOLLOW_UP_SURVEY_PROJECT"
								owner = event_target:country_of_interest
								location = this
							}					
						}
						75 = { }
					}
				}
			}
		}
	}
}

planet_event = {
	id = extended_land_survey.9
	is_triggered_only = yes
	location = ROOT	
	hide_window = yes

	immediate = {
		add_modifier = {
			modifier = underground_caverns
			days = -1
		}
		if = {
			limit = { planet_size < 6 }
			change_planet_size = 2
			break = yes
		}		
		if = {
			limit = { planet_size < 8 }
			change_planet_size = 3
			break = yes
		}
		if = {
			limit = { planet_size < 10 }
			change_planet_size = 4
			break = yes
		}
		if = {
			limit = { planet_size < 12 }
			change_planet_size = 5
			break = yes
		}
		if = {
			limit = { planet_size < 14 }
			change_planet_size = 6
			break = yes
		}
		if = {
			limit = { planet_size < 16 }
			change_planet_size = 7
			else = {
				change_planet_size = 8			
			}
		}	
	}
}
