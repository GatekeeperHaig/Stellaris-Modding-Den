@base_cost = 150
@city_cost = 250

district_settlement = {
	base_buildtime = 240
	is_capped_by_modifier = no

	show_on_uncolonized = {
		uses_district_set = colony
		exists = from
		from = { is_regular_empire = yes }
	}

	potential = {
		uses_district_set = colony
		exists = owner
		owner = { is_regular_empire = yes }
	}

	allow = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	conversion_ratio = 0.50
	convert_to = {
		district_city
		district_nexus
		district_hive
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = @city_cost
		}
		upkeep = {
			energy = 1
			consumer_goods = 1
		}
	}

	planet_modifier = {
		planet_housing_add = 2
		bio_pop_growth_from_immigration = 0.05
	}

	

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
			}
		}
		modifier = {
			job_colonist_add = 1
		}
	}
	triggered_desc = {
		text = job_colonist_effect_desc
	}
}
