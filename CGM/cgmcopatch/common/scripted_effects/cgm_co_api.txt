check_planet_bonus_buildings_co = {
	if = {
		limit = {
			NAND = {
				exists = owner
				owner = { has_authority = auth_machine_intelligence }
			}
		}
		 #UNIQUE
		if = {
			limit = { has_building = building_sacred_nexus }
			change_variable = { which = unity_mult_planet_building value = 0.10 }
		}
	}
	 #UNIQUE
	if = {
		limit = { has_building = building_research_coordination }
		change_variable = { which = physics_research_mult_planet_building value = 0.1 }
		change_variable = { which = society_research_mult_planet_building value = 0.1 }
		change_variable = { which = engineering_research_mult_planet_building value = 0.1 }
	}
	if = {
		limit = { has_building = building_food_bazaar }
		change_variable = { which = food_mult_planet_building value = 0.2 }
		change_variable = { which = energy_mult_planet_building value = 0.1 }
	}
	if = {
		limit = { has_building = building_colliseum }
		change_variable = { which = energy_mult_planet_building value = 0.15 }
	}
	 #NON_UNIQUE
	every_tile = {
		if = {
			limit = { has_building = building_resort }
			prev = {
				change_variable = { which = energy_mult_planet_building value = 0.1 }
			}
		}
	}
	if = {
		limit = { has_building = building_royal_library }
		change_variable = { which = society_research_mult_planet_building value = 0.25 }
	}
}
