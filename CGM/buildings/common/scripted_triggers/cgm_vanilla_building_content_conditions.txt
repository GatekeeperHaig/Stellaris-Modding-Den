 # Military Buildings
conditions_clone_vats = {
	planet = {
		is_habitat = no
		any_owned_pop = {
			is_robot_pop = no
			is_growing = no
			is_being_purged = no
			OR = {
				has_military_service_type = { type = military_service_full country = prev.owner }
				has_military_service_type = { type = military_service_limited country = prev.owner }
			}
		}
	}
}
conditions_slave_processing = {
	owner = {
		OR = { has_policy_flag = slavery_allowed is_hive_mind_empire = yes }
	}
	planet = {
		is_habitat = no
		any_owned_pop = {
			is_robot_pop = no
			is_growing = no
			is_being_purged = no
			is_enslaved = yes
			OR = {
				has_slavery_type = { country = from type = slavery_military }
				has_slavery_type = { country = from type = slavery_normal }
				has_slavery_type = { type = slavery_domestic }
			}
		}
	}
}
