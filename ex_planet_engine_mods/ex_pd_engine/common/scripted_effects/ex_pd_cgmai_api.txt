####it is stored in this mod, but is activated from the Core Game Mechanics and AI mod

check_planet_modifiers_pd = {
	if = {
		limit = { has_modifier = tidal_locked2  }
		change_variable = { which = energy_mult value = 0.10  }
		change_variable = { which = food_mult value = 0.10  }
	}
	if = {
		limit = { has_modifier = floating  }
		change_variable = { which = physics_research_mult value = 0.20  }
		change_variable = { which = energy_mult value = 0.20  }
	}
	if = {
		limit = { has_modifier = crystal  }
		change_variable = { which = unity_mult value = 0.20  }
		change_variable = { which = food_mult value = 0.20  }
	}
	if = {
		limit = { has_modifier = engineered_environment  }
		change_variable = { which = food_mult value = 0.10  }
		change_variable = { which = unity_mult value = 0.10  }
	}
}
