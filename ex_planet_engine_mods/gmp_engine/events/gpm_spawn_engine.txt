# initialisation called from gpm_game_start or the game start menu
# some spawn engine parts happen after 2 days to spread the workload on user's machines. This to reduce the initial "lag"
country_event = {
	id = gpm_spawn_engine.0
	hide_window = yes
	fire_only_once = yes
	is_triggered_only = yes
	
	immediate = {
		every_planet = {
			if = {
				limit = {
					gpm_is_planet_habitable = yes
					exists = owner
					is_fallen_empire_planet = yes
				}
				planet_event = { id = gpm_spawn_engine.1 days = 3}	# Fallen Empire Planet Modifiers + Effects
			}
			
			# Encounter regular planet: spawn modifiers
			if = {
				limit = {
					NOR = {
						gpm_is_planet_star = yes 
						gpm_is_planet_asteroid = yes
						gpm_is_planet_ringworld = yes
						AND = {
							is_capital = yes
							NOR = { 
								is_primitive_planet = yes
								is_presapient_planet = yes
							}
						}
						AND = {
							exists = owner
							is_fallen_empire_planet = yes
							OR = {
								is_colony = yes
								is_capital = yes
							}
						}
					}
				}
				planet_event = { id = gpm_spawn_engine.2 }	# Standard Planet Modifiers + Effects
			}
			
			# Encounter asteroid: spawn modifiers
			if = {
				limit = {
					gpm_is_planet_asteroid = yes
				}
				planet_event = { id = gpm_spawn_engine.3 days = 1 }	# Asteroid Modifiers + Effects
			}
			
			# Encounter star: spawn modifiers
			if = {
				limit = {
					gpm_is_planet_star = yes
				}
				planet_event = { id = gpm_spawn_engine.4 days = 2 }	# Star Modifiers + Effects
			}

			# Encounter ringworld: spawn modifiers
			if = {
				limit = {
					gpm_is_planet_ringworld = yes
					solar_system = {
						NOT = { exists = space_owner }
					}
				}
				planet_event = { id = gpm_spawn_engine.6 days = 3 }	# Ringworld Modifiers + Effects
			}

			# check to see and remove modifiers if these mods are present; Can be done cleaner but for now this stays
			# Further mod compatibility will be done later
			if = {
				limit = {
					OR = {
						STNH_enabled = yes
						new_worlds_enabled = yes
					}
				}
				planet_event = { id = gpm_spawn_engine.7 days = 1 } # Cleanup Planet Modifiers
			}
			

			# Check for Ego the living world system
			if = { limit = { solar_system = { has_star_flag = gpm_guardians_ego_system } }
				planet_event = { id = gpm_guardian.1 days = 1 } # spawn Ego
			}
		}

		# Further initialisations:
		country_event = { id = gpm_spawn_engine.5 }	# Spawn Natural Planet Wonders vanilla worlds
	}
}

# Fallen Empire Planet Modifiers + Effects
planet_event = {
	id = gpm_spawn_engine.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				is_capital = yes
			}
			# Remove all bad planet modifiers
			gpm_remove_negative_modifiers = yes
			# Spawn two guaranteed fallen empire modifier for capital worlds
			gpm_spawn_fallen_empire_modifier = yes
			gpm_spawn_fallen_empire_modifier = yes
			# Chance to spawn one fallen empire modifier
			random_list = {
				50 = {}
				50 = { gpm_spawn_fallen_empire_modifier = yes }
			}
		}
		if = {
			limit = {
				NOT = { is_capital = yes }
				is_colony = yes
			}
			# Remove all bad planet modifiers
			gpm_remove_negative_modifiers = yes
			# Spawn one guaranteed fallen empire modifier for colonies
			gpm_spawn_fallen_empire_modifier = yes
			random_list = {
				50 = {}
				50 = { gpm_spawn_fallen_empire_modifier = yes }
			}
			# Chance to spawn one fallen empire modifier
			random_list = {
				50 = {}
				50 = { gpm_spawn_fallen_empire_modifier = yes }
			}
		}
	}
}

					1 = { 
						gpm_spawn_precursor_modifier = yes

				has_global_flag = gpm_modifier_amount_high
			}
			if = { limit = { NOT = { num_modifiers > 3} }
				random_list = {

			}

			limit = {
				OR = {
					gpm_is_planet_barren = yes
					gpm_is_planet_frozen = yes
					gpm_is_planet_molten = yes

					90 = { gpm_spawn_tile_blockers = yes }
				}
			}
		}

			while = {
				count = 10
				random_list = {
					10 = { }
					90 = { gpm_spawn_precursor_tileblockers = yes }
				}
			}
		}
		if = { limit = { planet_size >= 15 planet_size < 20 gpm_has_precursor_modifiers = yes }
			while = {
				count = 7
				random_list = {
					10 = { }
					90 = { gpm_spawn_precursor_tileblockers = yes }
				}
			}
		}
		if = { limit = { planet_size < 15 gpm_has_precursor_modifiers = yes }
			while = {
				count = 5
				random_list = {
					10 = { }
					90 = { gpm_spawn_precursor_tileblockers = yes }
				}
			}
		}	
		
		# Cleanup some tileblockers if the planet is full
		gpm_cleanup_tileblockers = yes
		
		# Almost Empty planets we force spawn some:
		if = {
			limit = {
				count_tile = {
					limit = { has_blocker = yes }
					count < 5
				}	
			}
			if = { limit = { planet_size >= 20 }
				while = {
					count = 10
					random_list = {
						25 = { }
						75 = { gpm_spawn_vanilla_tileblockers = yes }
					}
				}
			}
			if = { limit = { planet_size >= 15 planet_size < 20 }
				while = {
					count = 7
					random_list = {
						25 = { }
						75 = { gpm_spawn_vanilla_tileblockers = yes }
					}
				}
			}
			if = { limit = { planet_size < 15 }
				while = {
					count = 5
					random_list = {
						25 = { }
						75 = { gpm_spawn_vanilla_tileblockers = yes }
					}
				}
			}
		}
	}
}
