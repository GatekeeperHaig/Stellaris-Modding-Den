
namespace = moonbase

planet_event = {
	id = moonbase.1
	hide_window = yes
	
	
    
	
trigger = {
	OR = {
		AND = {
			has_moon = yes 
			has_owner = yes
			NOT = { has_planet_flag = moon_habitat }}
			
		AND = {
			has_moon = yes
			any_moon = { has_owner = yes }
			NOT = { has_planet_flag = moon_habitat }
			}
			}	
		}

                    
                
               
		

immediate = {
	every_moon = {
		if = {
	        limit = { 
				has_owner = no
				NOT = { has_planet_flag = moon_habitat } }
	        
	    	
        set_planet_flag = moon_habitat

	}
}
}
}
