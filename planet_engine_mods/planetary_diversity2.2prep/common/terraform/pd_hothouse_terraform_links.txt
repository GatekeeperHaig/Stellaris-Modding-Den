#Machine

terraform_link = {
	to = "pc_machine"
	from = "pc_hothouse"
	
	energy = 20000
	duration = 10080
	
	potential = {
		has_ascension_perk = ap_machine_worlds
		from = { has_modifier = terraforming_candidate }
	}
	
	ai_weight = {
		weight = 20
		modifier = {
			factor = 0
			from = {
				any_owned_pop = {
					NOR = {
						has_trait = trait_machine_unit
						has_trait = trait_mechanical
					}
				}
			}
		}
		modifier = {
			factor = 0
			from = {
				is_colony = no
			}
		}
	}
}

#Gaia

terraform_link = {
	to = "pc_gaia"
	from = "pc_hothouse"
	
	energy = 20000
	duration = 14600
	
	potential = {
		has_ascension_perk = ap_machine_worlds
		from = { has_modifier = terraforming_candidate }
	}
	
	effect = {
		from = { remove_modifier = terraforming_candidate }
		from = { remove_modifier = bleak_planet }
		from = { remove_modifier = high_gravity }
		from = { remove_modifier = irradiated_planet }
		from = { remove_modifier = mineral_poor }
		from = { remove_modifier = unstable_tectonics }
		from = { remove_modifier = weak_magnetic_field }
		from = { remove_modifier = hazardous_weather }
		from = { remove_modifier = wild_storms }
		from = { remove_modifier = dangerous_wildlife }

	}
	
	ai_weight = {
		weight = 1
		modifier = {
			factor = 0
			OR = {
				is_mechanical_empire = yes
				is_cyborg_empire = yes
				has_authority = auth_machine_intelligence
			}
		}
	}
}
