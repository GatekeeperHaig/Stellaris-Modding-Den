namespace = pdanomaly

### Hothouse Terraforming Candidate
ship_event = {
	id = pdanomaly.10
	title = "pd.10.name"
	desc = {
		text = pd.10.desc.1
		trigger = { 
			NOT = { 
				owner = { has_technology = "tech_climate_restoration" } 
			}
		}
	}
	desc = {
		text = pd.10.desc.2
		trigger = { 
			owner = { has_technology = "tech_climate_restoration" }
		}
	}
	picture = GFX_evt_toxic
	location = FROM
	
	is_triggered_only = yes
	
	option = {
		name = FASCINATING
		custom_tooltip = TERRAFORMING_CANDIDATE_EFFECT
		hidden_effect = {
			from = {
				add_modifier = {
					modifier = "terraforming_candidate"
					days = -1
				}
			}		
		}
	}
}
