country_event = {
	id = fleet_cap.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_country_flag = fleet_cap_modifier1
	}
	
	immediate = {
		IF = {
			limit = {
				num_pops > 50
			}
			add_modifier = "fleet_cap_modifier2"
			remove_modifier = "fleet_cap_modifier1"
			country_event = {
				id = fleet_cap.2
			}
		}
	}
}
