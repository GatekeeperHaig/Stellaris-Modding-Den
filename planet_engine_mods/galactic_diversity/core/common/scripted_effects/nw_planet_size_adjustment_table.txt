new_worlds_adjust_asteroids_size = {
	if = {
		limit = { is_planet_class = pc_asteroid planet_size > 1 }
		random_list = {
			15 = { set_planet_size = 4 }
			25 = { set_planet_size = 3 }
			35 = { set_planet_size = 2 }
			25 = { set_planet_size = 1 }
		}	
	}
	if = {
		limit = { is_planet_class = pc_ice_asteroid planet_size > 1 }
		random_list = {
			25 = { set_planet_size = 3 }
			50 = { set_planet_size = 2 }
			25 = { set_planet_size = 1 }
		}	
	}
	if = {
		limit = { is_planet_class = pc_crystal_asteroid planet_size > 1 }
		random_list = {
			25 = { set_planet_size = 4 }
			50 = { set_planet_size = 3 }
			25 = { set_planet_size = 2 }
		}	
	}
}
