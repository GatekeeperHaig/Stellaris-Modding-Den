special_project = {
	key = "AURANTIACUM_PROJECT"
	cost = 2500
	tech_department = physics_technology
	icon = "gfx/interface/icons/situation_log/situation_log_planet.dds"
	picture = GFX_evt_alien_nature
	event_scope = planet_event
	on_success = {
		owner = { 
			set_country_flag = has_aurantiacum 
		}
		root = { 
			random_tile = { 
				limit = { has_resource = { type = sr_aurantiacum_x amount > 0 } }
				remove_blocker = yes
				set_building = building_aurantiacum_x_stabilizer
			}	
		}
	}
}

	event_scope = planet_event
	on_success = {
		owner = { 
			set_country_flag = has_ever_blossom
		}
		root = { 
			random_tile = { 
				limit = { has_resource = { type = sr_ever_blossom amount > 0 } }
				remove_blocker = yes
				if = { 
					limit = { prev.owner = { is_gestalt_consciousness = no } }
					set_building = building_lifetap
					else = { set_building = building_lifetap_gestalt }
				}	
			}	
		}
	}
