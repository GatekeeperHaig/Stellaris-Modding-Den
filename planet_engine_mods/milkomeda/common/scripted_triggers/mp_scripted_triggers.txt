# Returns yes if the system has any hyperlane to another system
has_any_hyperlane = {
	any_system = {
		has_hyperlane_to = PREV
	}
}

# Returns yes if the system has room for a hyperlane to another system
has_hyperlane_room = {
	any_neighbor_system = { 
		ignore_hyperlanes = yes
		NOT = {
			has_hyperlane_to = PREV
		}
	}
}

# Returns yes if the system has only one hyperlane to another system
has_one_hyperlane = {
	count_system = {
		count = 1
		limit = { 
			NOT = {
				is_same_value = PREV
			}
			has_hyperlane_to = PREV
		}
	}	
}

# Priority system check
is_priority_system = {
	OR = {
		exists = space_owner
		has_star_flag = hostile_system
		any_ship_in_system = { 
			always = yes 
		}
	}
}
