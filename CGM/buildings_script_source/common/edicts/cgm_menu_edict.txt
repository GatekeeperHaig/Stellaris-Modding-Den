country_edict = {
	name = cgm_menu_edict
	length = 0
	cost = { }

		hidden_effect = {
			country_event = { id = core_game_mechanics_and_ai_base.1 }
		}

		is_ai = no
		OR = {
			is_multiplayer = no

			}	
		}	
	}
}

country_edict = {
	name = cgm_enable_player_autobuild
	length = 0
	cost = { }
	potential = {
		is_ai = no
		OR = {
			has_country_flag = cgm_ai_disable_autobuild
			has_country_flag = cgm_sector_autobuild
		}
	}
	effect = {
		hidden_effect = {
			remove_country_flag = cgm_ai_disable_autobuild

		}
	}
}
