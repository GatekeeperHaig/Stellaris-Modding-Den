
@distance = 50
@base_moon_distance = 10
@jumps = 3

# Tidally-Locked World
pd_tidallylocked_start_init = {
	class = "sc_m"
	flags = { empire_home_system }
	usage = custom_empire

	init_effect = {
		every_neighbor_system = {
			set_star_flag = empire_cluster
			every_neighbor_system = {
				set_star_flag = empire_cluster
			}
		}
		generate_home_system_resources = yes
	}

	planet = {
		count = 1
		class = "pc_m_star"
		orbit_distance = 0
		orbit_angle = 1
		size = { min = 20 max = 30 }
		has_ring = no
	}
	
	change_orbit = 45
	
	planet = {
		count = { min = 0 max = 1 }
		orbit_distance = 20
		orbit_angle = { min = 90 max = 270 }
		class = rl_unhabitable_planets
		
		change_orbit = @base_moon_distance
		
		moon = {
			count = { min = 0 max = 1 }
			orbit_angle = { min = 90 max = 270 }
			orbit_distance = 5
			class = rl_unhabitable_planets
		}
	}
	
	change_orbit = 20

    planet = {
		class = "pc_tidallylocked"
		orbit_distance = 0
		orbit_angle = 85
		size = { min = 18 max = 21 }
		modifiers = none
		starting_planet = yes
		
		init_effect = {
			IF = {
				limit = { is_planet_class = pc_tidallylocked }
				set_planet_flag = pd_tidallylocked_planet
				add_modifier = {
					modifier = tidal_locked2
					days = -1
				}
				add_deposit = d_nightside_mining_operations
			}
			IF = {
				limit = { is_planet_class = pc_eyeball }
				set_planet_flag = pd_eyeball_planet
				add_modifier = {
					modifier = pdeyeball
					days = -1
				}
				add_deposit = d_eyeball_site
			}
        }
    }
            	
	change_orbit = 20
	
	planet = {
		count = { min = 6 max = 9 }
		orbit_distance = 20
		orbit_angle = { min = 90 max = 270 }
		class = rl_unhabitable_planets
		
		change_orbit = @base_moon_distance
		
		moon = {
			count = { min = 0 max = 1 }
			orbit_angle = { min = 90 max = 270 }
			orbit_distance = 5
			class = rl_unhabitable_planets
		}
	}
	
	neighbor_system = {
		trigger = { num_guaranteed_colonies >= 2 }
		hyperlane_jumps = { min = 1 max = @jumps }
		initializer = "neighbor_t1"
	}
	neighbor_system = {
		trigger = { num_guaranteed_colonies >= 1 }
		hyperlane_jumps = { min = 1 max = @jumps }
		initializer = "neighbor_t1_first_colony"
	}
}
