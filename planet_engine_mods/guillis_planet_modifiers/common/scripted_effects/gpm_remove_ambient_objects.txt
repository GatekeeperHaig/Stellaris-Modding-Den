# Planet Scope

gpm_remove_precursor_shields = {
	every_ambient_object = {
		if = {
			limit = { has_ambient_object_flag = gpm_precursor_shield_flag }
			destroy_ambient_object = this
			prev = { remove_planet_flag = gpm_precursor_shield_ambient }
		}
	}
}
